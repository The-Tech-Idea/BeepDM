Overview

EntityDataMoveValidator is a static utility class providing validation checks for moving records between entities and data sources. It validates:

  • Null checking — Required field validation
  • Uniqueness — Duplicate detection per field
  • Parent existence — Foreign key validation
  • Entity existence — Target entity availability
  • Structure comparison — Diff two entity structures
  • Data consistency — Cross-datasource data validation

Validation Methods

CanInsertRecord

Full record-level validation checking all constraints:

var result = EntityDataMoveValidator.CanInsertRecord(
    editor, ds, entity, record, "Customers", "MyDb");

switch (result)
{
    case EntityValidatorMesseges.OK:
        ds.InsertEntity("Customers", record);
        break;
    case EntityValidatorMesseges.NULL:
        Console.WriteLine("Required field is null");
        break;
    case EntityValidatorMesseges.NOT_UNIQUE:
        Console.WriteLine("Duplicate value detected");
        break;
    case EntityValidatorMesseges.PARENT_ERROR:
        Console.WriteLine("Referenced parent does not exist");
        break;
}

TrueifNotNull

var result = EntityDataMoveValidator.TrueifNotNull(fieldValue);
if (result != EntityValidatorMesseges.OK)
    Console.WriteLine("Field must not be null");

TrueifNotUnique

var result = EntityDataMoveValidator.TrueifNotUnique(
    editor, ds, entity, record, "Email", emailValue);
if (result != EntityValidatorMesseges.OK)
    Console.WriteLine("Email already exists");

TrueifParentExist

var result = EntityDataMoveValidator.TrueifParentExist(
    editor, ds, entity, record, "CustomerId", customerId);
if (result == EntityValidatorMesseges.PARENT_ERROR)
    Console.WriteLine($"Parent record {customerId} not found");

TrueifEntityExist

bool exists = EntityDataMoveValidator.TrueifEntityExist(
    editor, ds, entity);
if (!exists)
    Console.WriteLine("Target entity does not exist in datasource");

Structure Comparison

var sourceStruct = ds1.GetEntityStructure("Customers", false);
var targetStruct = ds2.GetEntityStructure("Customers", false);

var diffs = EntityDataMoveValidator.CompareEntityStructuresToList(
    sourceStruct, targetStruct,
    "SourceDB", "TargetDB");

foreach (var diff in diffs)
{
    Console.WriteLine($"[{diff.Severity}] {diff.FieldName}: {diff.Description}");
    // [ERROR] Email: Type mismatch (varchar(100) vs nvarchar(255))
    // [WARNING] Phone: Field missing in target
    // [INFO] Id: Types match
}

Usage Examples

Pre-Insert Validation Pipeline

public IErrorsInfo SafeInsert(IDMEEditor editor,
    string dsName, string entityName, object record)
{
    var ds = editor.GetDataSource(dsName);
    var entity = ds.GetEntityStructure(entityName, false);

    // Validate
    var result = EntityDataMoveValidator.CanInsertRecord(
        editor, ds, entity, record, entityName, dsName);

    switch (result)
    {
        case EntityValidatorMesseges.OK:
            return ds.InsertEntity(entityName, record);
        case EntityValidatorMesseges.NULL:
            return new ErrorsInfo { Flag = Errors.Failed,
                Message = "Required field is null" };
        case EntityValidatorMesseges.NOT_UNIQUE:
            return new ErrorsInfo { Flag = Errors.Failed,
                Message = "Duplicate value" };
        case EntityValidatorMesseges.PARENT_ERROR:
            return new ErrorsInfo { Flag = Errors.Failed,
                Message = "Parent record not found" };
        default:
            return new ErrorsInfo { Flag = Errors.Failed,
                Message = "Validation failed" };
    }
}

Migration Pre-Flight Check

// Before migration, compare structures
var sourceStruct = sourceDs.GetEntityStructure("Customers", false);
var targetStruct = targetDs.GetEntityStructure("Customers", false);

var diffs = EntityDataMoveValidator.CompareEntityStructuresToList(
    sourceStruct, targetStruct, "LegacyDB", "NewDB");

var hasErrors = diffs.Any(d => d.Severity == "ERROR");
if (hasErrors)
{
    Console.WriteLine("Structure mismatch - review before migration:");
    Console.WriteLine(EntityDataMoveValidator.GenerateReportFromComparison(diffs));
    return;
}

Console.WriteLine("Structures compatible - proceeding with migration");
