Overview

The Setup Framework automates the end-to-end initialization of BeepDM applications. It provides:

  • Step-based execution - Each step handles one responsibility (drivers, connections, schema, seeding)
  • Idempotency - Steps detect when work is done and skip safely
  • Checkpointing - State persisted after each step for resume after interruption
  • Cross-platform - Adapters for WinForms, WPF, Blazor, MAUI, CLI, Web API
  • Progress reporting - IProgress<PassedArgs> per-step
  • SHA-256 verified reports - Tamper-evident outcome with per-step timings

Core orchestrator that runs registered steps in sequence:

        string wizardId,

    public IErrorsInfo Run(
        SetupContext context,
        IProgress<PassedArgs> progress = null);

    // Resume from checkpoint
    public IErrorsInfo Resume(
        SetupContext context,
        IProgress<PassedArgs> progress = null);

    // Report
    public SetupReport GetReport();
}

Execution Flow

  1. Load persisted checkpoint if state is empty
  2. Validate step definitions (unique IDs, valid dependencies)
  3. For each incomplete step:
    • Check dependencies are completed
    • Validate step configuration
    • Check CanSkip (idempotency)
    • Execute step
    • Mark as completed
    • Persist checkpoint
  4. Build SHA-256 verified report

public class DefaultSetupWizardFactory : ISetupWizardFactory
{
    public (ISetupWizard wizard, SetupContext context) CreateDefault(IDMEEditor editor)
    {
        var options = new SetupOptions();
        var context = new SetupContext { Editor = editor, Options = options };
        var wizard = new SetupWizardBuilder().WithId("default-setup")
            .WithOptions(options).Build();
        return (wizard, context);
    }
}

Checkpointing & Resume

The framework automatically checkpoints state after each step:

// State file (JSON)
{
  "wizardId": "app-setup",
  "completedStepIds": ["driver-provision", "connection-config"],
  "skippedStepIds": [],
  "failedStepId": null,
  "completedSeederIds": [],
  "startedAt": "2024-01-15T10:30:00Z",
  "lastUpdatedAt": "2024-01-15T10:35:00Z"
}

Resume After Failure

// First run — fails at schema setup
var result1 = wizard.Run(context);
// State persisted with FailedStepId = "schema-setup"

// Fix the issue, then resume
var result2 = wizard.Resume(context);
// Loads checkpoint, skips completed steps, continues from schema-setup

Report Verification

var report = wizard.GetReport();
Console.WriteLine($"Success: {report.Succeeded}");
Console.WriteLine($"Duration: {report.FinishedAt - report.StartedAt}");
Console.WriteLine($"SHA-256: {report.ContentHash}");

foreach (var step in report.StepResults)
{
    Console.WriteLine($"  {step.StepName}: " +
        $"{(step.Succeeded ? "OK" : "FAIL")} " +
        $"({step.Elapsed.TotalSeconds:F1}s)");
}

Usage Examples

Desktop Application (WinForms)

var editor = new DMEEditor();
var connProps = new ConnectionProperties
{
    ConnectionName = "ProductDB",
    DatabaseType = DataSourceType.SqlLite,
    ConnectionString = "Data Source=./products.db"
};

var wizard = new SetupWizardBuilder()
    .WithId("product-app-setup")
    .WithEnvironment("Development")
    .AddStep(new DriverProvisionStep(new DriverProvisionStepOptions
    {
        PackageName = "TheTechIdea.Beep.DataSources.SQLite"
    }))
    .AddStep(new ConnectionConfigStep(new ConnectionConfigStepOptions
    {
        ConnectionProperties = connProps,
        OpenConnection = true
    }))
    .AddStep(new SchemaSetupStep(new SchemaSetupStepOptions
    {
        EntityTypes = new[] { typeof(Product), typeof(Category) }
    }))
    .AddStep(new SeedingStep(new SeedingStepOptions
    {
        Registry = seederRegistry
    }))
    .Build();

var adapter = new DesktopSetupWizardAdapter(
    onProgress: args => progressLabel.Text = args.Messege,
    onComplete: report => MessageBox.Show(
        report.Succeeded ? "Setup OK" : "Setup failed"));

var context = new SetupContext
{
    Editor = editor,
    Options = new SetupOptions
    {
        StateFilePath = "./setup-checkpoint.json"
    }
};

var report = await adapter.RunAsync(wizard, context);

Web API

[ApiController]
[Route("api/[controller]")]
public class SetupController : ControllerBase
{
    [HttpPost]
    public async Task<IActionResult> RunSetup()
    {
        var wizard = /* build wizard */;
        var adapter = new WebSetupWizardAdapter(
            hubContext: _hubContext);

        var context = new SetupContext { Editor = _editor };
        var report = await adapter.RunAsync(wizard, context);

        return Ok(new
        {
            Success = report.Succeeded,
            Steps = report.StepResults.Select(s => new
            {
                s.StepName,
                s.Succeeded,
                s.Elapsed
            })
        });
    }
}

CLI Application

var adapter = new CliSetupWizardAdapter(

Custom Step

public class CustomConfigStep : ISetupStep
{
    public string StepId => "custom-config";
    public string StepName => "Custom Configuration";
    public string Description => "Applies custom app settings";
    public IReadOnlyList<string> DependsOn => new[] { "connection-config" };

    public IErrorsInfo Validate(SetupContext context)
    {
        if (context.Editor == null)
            return new ErrorsInfo { Flag = Errors.Failed, Message = "Editor required" };
        return new ErrorsInfo { Flag = Errors.Ok };
    }

    public bool CanSkip(SetupContext context)
    {
        // Check if already configured
        return File.Exists("./custom-config.json");
    }

    public IErrorsInfo Execute(SetupContext context, IProgress<PassedArgs> progress)
    {
        // Apply custom configuration
        File.WriteAllText("./custom-config.json", "{}");
        progress?.Report(new PassedArgs { Messege = "Custom config applied" });
        return new ErrorsInfo { Flag = Errors.Ok };
    }
}

SetupReport & SetupStepResult

public class SetupReport
{
    public string WizardId { get; set; }                     // From builder
    public string RunId { get; set; } = Guid.NewGuid()       // Unique run ID
    public bool Succeeded { get; set; }                      // All steps OK
    public IReadOnlyList<SetupStepResult> StepResults { get; set; }
    public DateTimeOffset StartedAt { get; set; }
    public DateTimeOffset FinishedAt { get; set; }
    public TimeSpan TotalElapsed => FinishedAt - StartedAt;  // Computed
    public string ContentHash { get; set; }                  // SHA-256 of StepResults
    public string RollbackReportJson { get; set; }           // Phase 6
    public string DryRunReportJson { get; set; }             // Phase 3/8
    public string Environment { get; set; }                  // Target environment
}

public class SetupStepResult
{
    public string StepId { get; set; }
    public string StepName { get; set; }
    public bool Succeeded { get; set; }
    public bool Skipped { get; set; }
    public string Message { get; set; }
    public TimeSpan Elapsed { get; set; }
    public DateTimeOffset ExecutedAt { get; set; }
}
