Overview

NuGetPackageManager provides complete NuGet package lifecycle management with source management, caching, security validation, and shared context loading. It serves as the primary package management layer used by the AssemblyHandler for driver installation.

Usage Examples

Basic Package Management

var manager = new NuGetPackageManager(logger, loadContext);

// Search
var results = await manager.SearchAsync("postgresql", new SearchOptions
    { Take = 10, IncludePrerelease = false });

// Install
var result = await manager.InstallAsync("Npgsql", "7.0.0");
if (result.Success)
    Console.WriteLine($"Installed to: {result.InstallPath}");

// Load
var nugget = await manager.LoadAsync("Npgsql", "7.0.0");
Console.WriteLine($"Loaded: {nugget.Assemblies.Count} assemblies");

// Check
bool installed = await manager.IsInstalledAsync("Npgsql");
bool loaded = await manager.IsLoadedAsync("Npgsql");

Source Management

var sources = manager.GetActiveSources();
Console.WriteLine($"Active sources: {sources.Count}");

foreach (var src in sources)
    Console.WriteLine($"  {src.Name}: {src.Url} (Enabled: {src.IsEnabled})");

var url = manager.GetActiveSourceUrls();
Console.WriteLine($"Source URLs: {string.Join(", ", url)}");

Package Statistics

var stats = await manager.GetStatisticsAsync("Npgsql");
Console.WriteLine($"Installed: {stats.IsInstalled}");
Console.WriteLine($"Version: {stats.InstalledVersion}");
Console.WriteLine($"Last checked: {stats.LastChecked}");

// Get all installed packages
var installed = await manager.GetInstalledPackagesAsync();
foreach (var pkg in installed)
    Console.WriteLine($"{pkg.PackageId} v{pkg.Version}");
