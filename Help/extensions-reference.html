// EntityDiff result type
public class EntityDiff
{
    public IReadOnlyList<EntityField> Added { get; }
    public IReadOnlyList<EntityField> Removed { get; }
    public IReadOnlyList<FieldChange> Changed { get; }
    public bool HasChanges => Added.Count > 0 || Removed.Count > 0 || Changed.Count > 0;
}

// Individual field change detail
public class FieldChange
{
    public string FieldName { get; }
    public EntityField From { get; }      // Original field
    public EntityField To { get; }        // Changed field
    public string Summary { get; }        // Human-readable diff summary
}

FieldMergeStrategy

StrategyBehaviorUse Case
KeepExistingNew fields are added; existing fields are never overwrittenSafe schema sync — don't change existing
OverwriteExisting fields replaced entirely by source fieldsSchema migration — source is authoritative
PreferNonNullOnly overwrites null/empty properties; merges non-null valuesIncremental enrichment from partial schemas

BatchExtensions

Efficient batching of IEnumerable<T> with lazy streaming — no intermediate materialization.

public static class BatchExtensions
{
    // Split into batches of batchSize
    public static IEnumerable<IEnumerable<T>> Batch<T>(this IEnumerable<T> source, int batchSize);

    // Split + project per batch (read-only projection to avoid mutation)
    public static IEnumerable<TResult> Batch<T, TResult>(
        this IEnumerable<T> source, int batchSize,
        Func<IReadOnlyList<T>, TResult> resultSelector);

    // Read-only batches to prevent external modification
    public static IEnumerable<ReadOnlyCollection<T>> ToSafeBatches<T>(
        this IEnumerable<T> source, int batchSize);

    // Async batch processing with optional concurrency
    public static async IAsyncEnumerable<TResult> BatchForEachAsync<T, TResult>(
        this IEnumerable<T> source, int batchSize,
        Func<IReadOnlyList<T>, CancellationToken, Task<TResult>> processor,
        int maxConcurrency = 1, CancellationToken ct = default);
}

Usage

// Simple batching
foreach (var batch in records.Batch(100))
    await InsertBatchAsync(batch);

// Batch with projection
var summaries = records.Batch(100, batch => new
{
    Count = batch.Count,
    First = batch[0],
    Last = batch[^1]
});

// Async with concurrency
await foreach (var result in records.BatchForEachAsync(
    100, async (batch, ct) => { await bulkInsertAsync(batch, ct); return batch.Count; },
    maxConcurrency: 4))
{
    Console.WriteLine($"Inserted {result} records");
}
