NuGet Plugin Loading

// Configuring SharedContextAssemblyHandler for NuGet
var handler = new SharedContextAssemblyHandler(config, errors, logger, util);
var nugetManager = handler.NuGetService;

// Download and load as plugin
var downloader = new NuggetPackageDownloader();
string packagePath = await downloader.DownloadAsync(
    "MyCompany.Plugin", "1.0.0", "./plugins/");

var loader = new NuggetPluginLoader();
var manifest = await loader.LoadPluginManifestAsync(
    "MyCompany.Plugin", "1.0.0");

// Register in the plugin system
PluginRegistry.Register(manifest);
PluginLifecycleManager.InstallPlugin(manifest);

// Track in shared context
SharedContextManager.LoadDll(packagePath);
SharedAssemblyTracker.TrackAssembly(assembly, packagePath);

Usage Examples

Using SharedContextAssemblyHandler

var handler = new SharedContextAssemblyHandler(
    config, errors, logger, util);

// Load all assemblies with progress
var progress = new Progress<PassedArgs>(p =>
    Console.WriteLine($"Loading: {p.Messege}"));
handler.LoadAllAssembly(progress, CancellationToken.None);

// Results
Console.WriteLine($"Assemblies: {handler.LoadedAssemblies.Count}");
Console.WriteLine($"Data sources: {handler.DataSourcesClasses.Count}");
Console.WriteLine($"Addins: {handler.AddinClasses.Count}");

Plugin Lifecycle

var lifecycle = new PluginLifecycleManager(logger);
lifecycle.PluginStateChanged += (s, e) =>
    Console.WriteLine($"Plugin {e.PluginId}: {e.NewState}");

// Register
lifecycle.RegisterPlugin(new PluginInfo { PluginId = "my-plugin" });

// Install
var manifest = new PluginManifest
{
    PluginId = "my-plugin",
    Version = new Version(1, 0, 0),
    EntryTypes = new List<string> { "MyPlugin.MyAddin" }
};
await lifecycle.InstallPlugin(manifest);

// Start
await lifecycle.StartPlugin("my-plugin");
var state = lifecycle.GetPluginState("my-plugin"); // PluginState.Running

// Monitor
var health = new PluginHealthMonitor();
health.TrackPlugin("my-plugin", heartbeatIntervalMs: 3000);
health.HealthStatusChanged += (s, e) =>
    Console.WriteLine($"Health: {e.Status.State}");

Isolated Plugin Loading

var isolation = new PluginIsolationManager();
var ctx = isolation.CreateIsolatedContext("sandboxed-plugin");

var asm = isolation.LoadAssemblyInContext("sandboxed-plugin", "./plugins/sandbox.dll");
var instance = isolation.CreateInstanceInContext("sandboxed-plugin",
    "SandboxPlugin.Worker");

// Can unload the entire context
isolation.UnloadIsolatedContext("sandboxed-plugin");
// All assemblies in that ALC are released
