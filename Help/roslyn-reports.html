RoslynCompiler

Static compiler using Microsoft.CodeAnalysis for runtime C# compilation, dynamic assembly generation, and type caching.

public static class RoslynCompiler
{
    // Type cache: class name -> (Type, Assembly)
    private static readonly Dictionary<string, Tuple<Type, Assembly>> CompiledTypes;

    // File-to-DLL compilation
    public static bool CompileCodeToDLL(string sourceFile, string outputFile);
    public static bool CompileCodeToDLL(IEnumerable<string> sourceFiles, string outputFile);
    public static bool CompileClassFromFileToDLL(string filePath, string outputFile);
    public static bool CompileClassFromStringToDLL(string sourceCode, string outputFile);

    // Type resolution
    public static Tuple<Type, Assembly> CompileClassTypeandAssembly(string classname, string code);
    public static Type CompileGetClassType(string filepath, string classname, string code);

    // In-memory assembly creation
    public static Assembly CreateAssembly(IDMEEditor DMEEditor, string code);
    public static Assembly CreateAssembly(string code);
    public static Task<Assembly> CreateAssemblyAsync(IDMEEditor DMEEditor, string code);

    // Diagnostics
    public static (bool Success, IEnumerable<Diagnostic> Errors, Assembly Assembly) CompileAndGetDiagnostics(string code);

    // Cache management
    public static void ClearCompiledTypeCache();
    public static bool RemoveFromCache(string className);

    // Advanced
    public static bool CompileWithDebuggingInfo(string sourceCode, string outputFile);
    public static Assembly CompileWithSourceGenerators(string code, IEnumerable<ISourceGenerator> gen);
    public static Dictionary<string, Type> CompileMultipleClassTypes(Dictionary<string, string> classes);
}

Advanced Features

// Compile with debugging info (PDB)
bool debugOk = RoslynCompiler.CompileWithDebuggingInfo(code, "output.pdb");

// Compile with source generators
var generators = new[] { new CustomSourceGenerator() };
Assembly asm = RoslynCompiler.CompileWithSourceGenerators(code, generators);

// Async compilation
Assembly asmAsync = await RoslynCompiler.CreateAssemblyAsync(editor, code);

// Cache management
RoslynCompiler.ClearCompiledTypeCache();
RoslynCompiler.RemoveFromCache("DynamicEntity");

// Error diagnostics
var (success, errors, assembly) = RoslynCompiler.CompileAndGetDiagnostics(code);
foreach (var diagnostic in errors.Take(10))
    Console.WriteLine($"[{diagnostic.Severity}] {diagnostic.GetMessage()}");

ReportDataManager

Report data aggregation engine. Reads IAppDefinition blocks, builds SQL queries against configured datasources, and populates report datasets.

public class ReportDataManager
{
    public ReportDataManager(IDMEEditor pDMEEditor);
    public ReportDataManager(IDMEEditor pDMEEditor, IAppDefinition pDefinition);

    public IAppDefinition Definition { get; set; }
    public IDMEEditor DMEEditor { get; set; }
    public DataSet Data { get; set; }

    // Report execution
    public DataSet GetDataSet();
    public string GetReportFilePath(IAppDefinition reportDefinition);

    // Data queries
    public object GetList(string EntityID);
    public object GetList(AppBlock blk);
}

How It Works

  1. Accepts IAppDefinition with block definitions and filters
  2. Iterates each block's columns to build SELECT field lists
  3. Builds SQL queries using block-level filters and child relations
  4. Executes queries against configured datasources
  5. Populates a DataSet with all report data

Usage Examples

Dynamic Entity Compilation

string code = @"
    public class DynamicCustomer
    {
        public int Id { get; set; }
        public string Name { get; set; }
        public string Email { get; set; }
    }";

var (type, assembly) = RoslynCompiler.CompileClassTypeandAssembly(
    "DynamicCustomer", code);

// Create instance
var customer = Activator.CreateInstance(type);
type.GetProperty("Name").SetValue(customer, "John");

// Use with UnitOfWork
var uow = UnitOfWorkFactory.CreateUnitOfWork(
    type, editor, "MyDB", "Customers");

Report Generation

var def = editor.ConfigEditor.LoadAppDefinition("SalesReport");
var reportManager = new ReportDataManager(editor, def);

// Generate report data
var dataSet = reportManager.GetDataSet();

// Render to file
var output = new ReportOutput();
output.Render(dataSet, "report.html", ReportType.Html);

Batch Compilation

var classes = new Dictionary<string, string>
{
    ["DynamicCustomer"] = customerCode,
    ["DynamicOrder"] = orderCode,
    ["DynamicProduct"] = productCode
};

var types = RoslynCompiler.CompileMultipleClassTypes(classes);
foreach (var (name, type) in types)
    Console.WriteLine($"Compiled: {name} -> {type.FullName}");
