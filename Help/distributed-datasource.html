Overview

DistributedDataSource distributes entities across multiple physical datasources (shards), each backed by an IProxyCluster. It implements IDataSource and IDistributedDataSource across 12 partial files.

Per-shard failover, circuit breaking, and load balancing are inherited from IProxyCluster.

DistributionMode Enum

public enum DistributionMode
{
    Routed    = 0,  // Entire entity on one shard, point reads/writes
    Sharded   = 1,  // Row-level sharding via partition function, scatter reads
    Replicated = 2, // Entity on every shard, fan-out writes, any-shard reads
    Broadcast = 3   // Replicated with closest-healthy-shard read preference
}

Usage Examples

Basic Setup

var shards = new Dictionary<string, IProxyCluster>
{
    ["shard-a"] = shardACluster,
    ["shard-b"] = shardBCluster,
    ["shard-c"] = shardCCluster
};

var plan = DistributionPlanBuilder.Create()
    .Shard("Customers", on: "CustomerId",
        across: new[] { "shard-a", "shard-b" })
    .Replicate("Products",
        across: new[] { "shard-a", "shard-b", "shard-c" })
    .Build();

var distributed = new DistributedDataSource(
    editor, shards, plan, options);

distributed.Openconnection();

Backpressure Handling

try
{
    var rows = distributed.GetEntity("Orders", null);
}
catch (BackpressureException bp)
{
    Console.WriteLine($"Retry after {bp.RetryAfter.TotalMilliseconds}ms");
}

Monitoring

distributed.OnShardDown += (s, e) =>
    logger.LogWarning($"Shard down: {e.ShardId}");

distributed.OnDegradedMode += (s, e) =>
    logger.LogWarning($"Degraded. Healthy ratio: {e.HealthyRatio:F2}");

distributed.OnPlacementViolation += (s, e) =>
    logger.LogError($"Placement violation: {e.EntityName}");
