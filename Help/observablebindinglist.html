Overview

ObservableBindingList<T> is the single source of truth collection used by UnitofWork<T>. It extends BindingList<T> and implements IBindingListView, INotifyCollectionChanged, and IDisposable.

Change Tracking

Dual-indexed tracking with O(1) lookup:

// O(1) lookup by Guid
private Dictionary<Guid, Tracking> _trackingsByGuid;

// O(1) lookup by original index
private Dictionary<int, Tracking> _trackingsByOriginalIndex;

// Tracking state per entity
public class Tracking
{
    public Guid UniqueId { get; set; }
    public int OriginalIndex { get; set; }
    public EntityState EntityState { get; set; } // Added, Modified, Deleted, Unchanged
    public string ModifiedBy { get; set; }
    public DateTime ModifiedAt { get; set; }
}

Usage

var list = new ObservableBindingList<Customer>(customers);

// Check tracking
foreach (var tr in list.Trackings)
{
    Console.WriteLine($"Item {tr.OriginalIndex}: {tr.EntityState} by {tr.ModifiedBy}");
}

// Get added entities
var added = list.Trackings.Where(t => t.EntityState == EntityState.Added);

// Get modified entities
var modified = list.Trackings.Where(t => t.EntityState == EntityState.Modified);

Sorting & Filtering

var list = new ObservableBindingList<Customer>(customers);

// Sort
list.ApplySort("Name", ListSortDirection.Ascending);
list.ApplySort("CreatedDate", ListSortDirection.Descending);

// Filter
list.Filter = "Name LIKE 'John*' OR City = 'NYC'";
list.Filter = "Status = 'Active'";

// Clear filter
list.RemoveFilter();

Pagination

list.PageSize = 50;

// Navigate pages
list.GoToPage(1);   // First page
list.GoToPage(3);   // Page 3
list.NextPage();
list.PreviousPage();

// Info
Console.WriteLine($"Page {list.CurrentPage} of {list.TotalPages}");
Console.WriteLine($"Items: {list.Count}");

Validation

// Data annotations on entity
public class Customer : INotifyPropertyChanged
{
    [Required, StringLength(100)]
    public string Name { get; set; }

    [Required, EmailAddress]
    public string Email { get; set; }

    [Range(0, 200)]
    public int Age { get; set; }
}

// Validate
var list = new ObservableBindingList<Customer>(customers);
var result = list.ValidateAll();
if (!result.IsValid)
{
    foreach (var error in result.Errors)
        Console.WriteLine($"{error.PropertyName}: {error.Message}");
}

Undo/Redo

list.EnableUndoRedo(maxDepth: 50);

customer.Name = "Changed";
list.Undo();  // Name reverts
list.Redo();  // Restores change

// Check state
if (list.CanUndo)
    list.UndoLastChange();

if (list.CanRedo)
    list.RedoLastChange();

list.ClearUndoHistory();

Virtual Loading

// Enable virtual loading for large datasets
list.EnableVirtualMode(totalCount: 100000);

await list.GoToPageAsync(5);
await list.PrefetchAdjacentPagesAsync();

// Page cache
list.PageCacheSize = 5; // Keep 5 pages cached
list.InvalidatePageCache();
list.DisableVirtualMode();
