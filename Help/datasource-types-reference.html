Capability API

// Get capabilities for a datasource type
var caps = DataSourceCapabilityMatrix.GetCapabilities(DataSourceType.SqlServer);

// String-based check (case-insensitive)
bool canJoin = caps.IsCapable("SupportsJoins");         // true
bool canVector = caps.IsCapable("SupportsVectorSearch"); // false

// Enum-based check (strongly typed)
bool canBulk = caps.IsCapable(CapabilityType.SupportsBulkOperations); // true

// CapabilityType enum: 36 values
// SupportsTransactions, SupportsJoins, SupportsAggregations,
// SupportsIndexes, SupportsIdentity, SupportsBulkOperations,
// SupportsVectorSearch, SupportsGraphTraversal, SupportsStreaming,
// SupportsPushNotifications, SupportsOAuth, ...

Connection Examples

RDBMS

var props = new ConnectionProperties
{
    ConnectionName = "MyDB",
    DatabaseType = DataSourceType.SqlServer,
    Category = DatasourceCategory.RDBMS,
    Host = "localhost",
    Port = 1433,
    Database = "ProductsDB",
    UserID = "app_user",
    Password = "secret"
};

MongoDB

var props = new ConnectionProperties
{
    ConnectionName = "Mongo",
    ConnectionString = "mongodb://localhost:27017",
    DatabaseType = DataSourceType.MongoDB,
    Category = DatasourceCategory.NOSQL,
    Database = "ProductCatalog"
};

File-Based

var props = new ConnectionProperties
{
    ConnectionName = "CSVData",
    ConnectionString = "./data/",
    FileName = "export.csv",
    DatabaseType = DataSourceType.Csv,
    Category = DatasourceCategory.FileBased
};

Web API

var props = new ConnectionProperties
{
    ConnectionName = "ExternalAPI",
    ConnectionString = "https://api.example.com/v1",
    DatabaseType = DataSourceType.WebService,
    Category = DatasourceCategory.CLOUD,
    Parameters = "Bearer token123"
};
