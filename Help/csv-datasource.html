Overview

CSVDataSource implements IDataSource for CSV files with automatic delimiter detection, type inference via table scanning, and full CRUD operations. It supports both CSV and XLS datasource types via a single implementation.

  • Automatic delimiter detection — Detects comma, tab, semicolon, or pipe delimiters
  • Type inference — Scans rows to determine field types (int, decimal, DateTime, etc.)
  • Auto-creation — Creates empty file if not found
  • File locking — Retry logic for locked files with exponential backoff
  • CSVTextFieldParser — Robust quoted-field parsing via Microsoft.VisualBasic

Constructor & Delimiter Detection

The constructor automatically detects the delimiter from the file:

// Delimiter detection logic
public CSVDataSource(string datasourcename, IDMLogger logger,
    IDMEEditor pDMEEditor, DataSourceType pDatasourceType, IErrorsInfo per)
{
    DatasourceName = datasourcename;
    Logger = logger;
    ErrorObject = per;
    DMEEditor = pDMEEditor;
    DatasourceType = pDatasourceType;
    Category = DatasourceCategory.FILE;

    // Create FileConnection
    Dataconnection = new FileConnection(DMEEditor);

    // Read config
    var connProp = DMEEditor.ConfigEditor.DataConnections
        .FirstOrDefault(c => c.FileName == datasourcename);
    Dataconnection.ConnectionProp = connProp;

    // Detect delimiter from file
    string fullPath = Path.Combine(connProp.FilePath, connProp.FileName);
    if (File.Exists(fullPath))
        Delimiter = DetectDelimiter(fullPath); // , ; \t |
    else
        Delimiter = ',';
}

Connection Lifecycle

public ConnectionState Openconnection()
{
    // Retry logic for file locks (5 attempts, exponential backoff)
    for (int i = 0; i < 5; i++)
    {
        try
        {
            Dataconnection.OpenConnection();
            if (Dataconnection.ConnectionStatus == ConnectionState.Open)
            {
                fieldParser = new CsvTextFieldParser(fullPath);
                fieldParser.Delimiters = new[] { Delimiter.ToString() };
                fieldParser.HasFieldsEnclosedInQuotes = true;
                return ConnectionState.Open;
            }
        }
        catch (IOException ex) when (i < 4)
        {
            Thread.Sleep(500 * (int)Math.Pow(2, i)); // Exponential backoff
        }
    }
    return ConnectionState.Closed;
}

public ConnectionState Closeconnection()
{
    fieldParser?.Close();
    fieldParser = null;
    return Dataconnection.CloseConn();
}

public ConnectionState GetFileState() => ConnectionStatus == ConnectionState.Open
    ? ConnectionStatus : Openconnection();

Field Scanning & Type Inference

// Scans up to 100 rows to determine column types
private List<EntityField> GetFieldsbyTableScan(
    string sheetname, List<EntityField> flds, string[] headerRow)
{
    int i = 0;
    while (fieldParser.EndOfData == false && i < 100)
    {
        string[] row = fieldParser.ReadFields();
        i++;
        for (int j = 0; j < flds.Count; j++)
        {
            EntityField f = flds[j];
            string val = row[j];

            if (f.FieldName.Contains("date") || f.FieldName.EndsWith("_dt"))
                f.Fieldtype = "System.DateTime";
            else if (int.TryParse(val, out _))
                f.Fieldtype = "System.Int32";
            else if (decimal.TryParse(val, out _))
                f.Fieldtype = "System.Decimal";
            else if (DateTime.TryParse(val, out _))
                f.Fieldtype = "System.DateTime";
            else if (bool.TryParse(val, out _))
                f.Fieldtype = "System.Boolean";
            else
                f.Fieldtype = "System.String";
        }
    }
    IsFieldsScanned = true;
    return flds;
}

Usage Examples

Basic CSV Access

var props = new ConnectionProperties
{
    ConnectionName = "Customers",
    ConnectionString = "./data/",
    FileName = "customers.csv",
    DatabaseType = DataSourceType.CSV,
    Category = DatasourceCategory.FILE
};

editor.ConfigEditor.AddDataConnection(props);
var ds = editor.GetDataSource("Customers");
ds.Openconnection();

// Read data
var customers = ds.GetEntity("customers", new List<AppFilter>());
foreach (var c in customers)
    Console.WriteLine(c);

// Get entity list
var entities = ds.GetEntitesList();
foreach (var name in entities)
    Console.WriteLine(name);

With Filters

var filtered = ds.GetEntity("customers", new List<AppFilter>
{
    new AppFilter { FieldName = "Country", Operator = "=", FilterValue = "USA" },
    new AppFilter { FieldName = "Status", Operator = "=", FilterValue = "Active" }
});
