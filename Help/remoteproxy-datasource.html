RemoteProxyDataSource

Routes all operations to a remote worker machine via IProxyTransport, enabling multi-machine cluster topologies

Overview

RemoteProxyDataSource routes every IDataSource operation to a remote worker machine via an IProxyTransport. All complex arguments are serialized as JSON before being sent over the wire; all complex return values are deserialized from the worker's JSON response.

Drop this into a ProxyNode inside a ProxyCluster to make the cluster topology span multiple machines.

Class Definition

public sealed class RemoteProxyDataSource : IProxyDataSource
{
    public RemoteProxyDataSource(
        IProxyTransport transport,
        string nodeId,
        IDMEEditor editor);

    // IDataSource identity
    public string GuidID { get; set; }
    public string DatasourceName { get; set; }
    public DataSourceType DatasourceType { get; set; }
    public DatasourceCategory Category { get; set; }
    public IErrorsInfo ErrorObject { get; set; }
    public IDMLogger Logger { get; set; }
    public ConnectionState ConnectionStatus { get; set; }
    public IDataConnection Dataconnection { get; set; }
    public IDMEEditor DMEEditor { get; set; }
    public List<string> EntitiesNames { get; set; }
    public List<EntityStructure> Entities { get; set; }
    public string ColumnDelimiter { get; set; }
    public string ParameterDelimiter { get; set; }

    // IProxyDataSource backward-compat knobs
    public int MaxRetries { get; set; } = 3;
    public int RetryDelayMilliseconds { get; set; } = 500;
    public int HealthCheckIntervalMilliseconds { get; set; } = 30_000;

    // Watchdog
    public int WatchdogIntervalMs { get; set; } = 5_000;
    public int WatchdogProbeTimeoutMs { get; set; } = 2_000;
    public int WatchdogFailureThreshold { get; set; } = 2;
    public int WatchdogRecoveryThreshold { get; set; } = 3;

    // Audit
    public bool EnableAudit { get; set; }
}

IProxyTransport

The transport abstraction routes calls to the remote worker:

public interface IProxyTransport
{
    Task<ProxyResult> SendAsync(ProxyRequest request);
}

// Built-in: HttpProxyTransport
var transport = new HttpProxyTransport("http://worker-b:5100");
var remote = new RemoteProxyDataSource(transport, "worker-b", editor);

Cluster Integration

// Create remote node
var transport = new HttpProxyTransport("http://worker-b:5100");
var remote = new RemoteProxyDataSource(transport, "worker-b", editor);

// Add to cluster
var cluster = new ProxyCluster(policy);
cluster.AddNode(new ProxyNode("worker-b", remote, weight: 2,
    role: ProxyDataSourceRole.Primary));

// With audit
remote.EnableAudit = true;
remote.MaxRetries = 5;
remote.RetryDelayMilliseconds = 1000;

Usage Examples

Multi-Machine Cluster

var policy = new ProxyPolicy
{
    NodeRoutingStrategy = NodeRoutingStrategy.FastestResponse
};

var cluster = new ProxyCluster(policy);

// Local node
cluster.AddNode(new ProxyNode("local", localProxy, weight: 1));

// Remote worker nodes
var workerA = new RemoteProxyDataSource(
    new HttpProxyTransport("http://worker-a:5100"), "worker-a", editor);
var workerB = new RemoteProxyDataSource(
    new HttpProxyTransport("http://worker-b:5100"), "worker-b", editor);

cluster.AddNode(new ProxyNode("worker-a", workerA, weight: 2,
    role: ProxyDataSourceRole.Primary));
cluster.AddNode(new ProxyNode("worker-b", workerB, weight: 1,
    role: ProxyDataSourceRole.Replica));

// Cluster auto-routes to fastest healthy node
var data = cluster.GetEntity("Customers", filters);
