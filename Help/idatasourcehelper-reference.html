Overview

IDataSourceHelper is the central abstraction enabling database-agnostic operations across 30+ datasource types. Each datasource (RDBMS, MongoDB, Redis, etc.) implements this interface to provide:

  • Unified API — Same method calls regardless of backend
  • Query generation — Native SQL/CQL/aggregation pipeline per datasource
  • Feature detection — Graceful degradation via DataSourceCapabilities
  • Type mapping — CLR types to datasource-specific types

Interface Definition

public interface IDataSourceHelper
{
    DataSourceType SupportedType { get; set; }
    string Name { get; }
    DataSourceCapabilities Capabilities { get; }

    // Schema Operations
    (string Query, bool Success) GetSchemaQuery(string userName);
    (string Query, bool Success) GetTableExistsQuery(string tableName);
    (string Query, bool Success) GetColumnInfoQuery(string tableName);

    // DDL Operations (Level 1)
    (string Sql, bool Success, string ErrorMessage) GenerateCreateTableSql(
        EntityStructure entity, string schemaName = null,
        DataSourceType? dataSourceType = null);
    // ... Drop, Truncate, Index, Column, Rename operations

    // Constraint Operations (Level 2)
    // ... Primary Key, Foreign Key, Unique, GetConstraints operations

    // Transaction Control (Level 3)
    (string Sql, bool Success) BeginTransaction();
    (string Sql, bool Success) CommitTransaction();
    (string Sql, bool Success) RollbackTransaction();

    // DML Operations (Level 4)
    // ... Insert, Update, Delete, Select with entity/parameter support

    // Utility Methods
    string QuoteIdentifier(string identifier);
    // ... Type mapping, data conversion helpers
}

Schema Operations

// List schemas/databases
var (sql, ok) = helper.GetSchemaQuery("app_user");
// SQL Server: SELECT name FROM sys.databases WHERE HAS_DBACCESS(name)=1
// PostgreSQL: SELECT schema_name FROM information_schema.schemata
// MySQL: SHOW DATABASES

// Check if table exists
var (sql, ok) = helper.GetTableExistsQuery("Customers");
// SQL Server: SELECT OBJECT_ID('Customers')
// PostgreSQL: SELECT EXISTS(SELECT FROM information_schema.tables...)
// SQLite: SELECT name FROM sqlite_master WHERE type='table'

// Get column info
var (sql, ok) = helper.GetColumnInfoQuery("Customers");
// SQL Server: sp_columns 'Customers'
// MySQL: SHOW COLUMNS FROM Customers

DDL Operations (Level 1)

var (sql, success, error) = helper.GenerateCreateTableSql(entity);
// SQL Server: CREATE TABLE [Customers] (Id INT IDENTITY PRIMARY KEY...)
// PostgreSQL: CREATE TABLE "Customers" (Id SERIAL PRIMARY KEY...)
// MySQL: CREATE TABLE `Customers` (Id INT AUTO_INCREMENT...)

var (sql, ok, err) = helper.DropTable(entity, "Customers", "dbo");
// SQL Server: DROP TABLE [dbo].[Customers]

var (sql, ok, err) = helper.TruncateTable(entity, "Customers");
// SQL Server: TRUNCATE TABLE [Customers]
// PostgreSQL: TRUNCATE TABLE "Customers" RESTART IDENTITY

var (sql, ok, err) = helper.CreateIndex(entity, "idx_email", "Email");
var (sql, ok, err) = helper.AddColumn(entity, "NewField", "VARCHAR(100)", null);
var (sql, ok, err) = helper.AlterColumn(entity, "OldField", "INT", false);
var (sql, ok, err) = helper.DropColumn(entity, "TempField");
var (sql, ok, err) = helper.RenameTable(entity, "OldName", "NewName");
var (sql, ok, err) = helper.RenameColumn(entity, "OldCol", "NewCol");

Constraint Operations (Level 2)

var (sql, ok) = helper.AddPrimaryKey(entity, "PK_Customers", "Id");
var (sql, ok) = helper.DropPrimaryKey(entity, "PK_Customers");

var (sql, ok) = helper.AddForeignKey(entity, "FK_CustomerId",
    "CustomerId", "Customers", "Id");
var (sql, ok) = helper.DropForeignKey(entity, "FK_CustomerId");

var (sql, ok) = helper.AddUniqueConstraint("Customers", "UQ_Email", "Email");
var (sql, ok) = helper.DropUniqueConstraint("Customers", "UQ_Email");

var (sql, ok) = helper.GetConstraints(entity, "Customers");

Transaction Control (Level 3)

var (sql, ok) = helper.BeginTransaction();
// SQL Server: BEGIN TRANSACTION
// PostgreSQL: BEGIN
// MySQL: START TRANSACTION

var (sql, ok) = helper.CommitTransaction();
// SQL Server: COMMIT TRANSACTION
// PostgreSQL: COMMIT

var (sql, ok) = helper.RollbackTransaction();
// SQL Server: ROLLBACK TRANSACTION
// PostgreSQL: ROLLBACK

DML Operations (Level 4)

// INSERT with parameter support
var (sql, success, error) = helper.Insert(
    entity, "Customers", new { Name = "John" });
// SQL Server: INSERT INTO [Customers] (Name) VALUES (@Name)
// PostgreSQL: INSERT INTO "Customers" ("Name") VALUES ($1)

// UPDATE
var (sql, ok, err) = helper.Update(entity, "Customers",
    new { Name = "Jane" }, new { Id = 1 });

// DELETE
var (sql, ok, err) = helper.Delete(entity, "Customers",
    new { Id = 1 });

// SELECT with filter
var (sql, ok, err) = helper.Select(entity, "Customers",
    new List<AppFilter> { new AppFilter { FieldName = "Status", FilterValue = "Active" } });

// Bulk operations
var (sql, ok, err) = helper.BulkInsert(entity, "Customers", records);
var (sql, ok, err) = helper.BulkUpdate(entity, "Customers", records);
var (sql, ok, err) = helper.BulkDelete(entity, "Customers", ids);

Utility Methods

// Identifier quoting per datasource
string quoted = helper.QuoteIdentifier("My Table");
// SQL Server: [My Table]
// PostgreSQL: "My Table"
// MySQL: `My Table`

// Type mapping
string dbType = helper.MapClrToDataSourceType("System.String", 100);
// SQL Server: NVARCHAR(100)
// PostgreSQL: VARCHAR(100)

// Data conversion (value, direction)
object converted = helper.ConvertData("123", "int", DataDirection.ToDataSource);

// Capability checking
if (helper.Capabilities.SupportsTransactions)
{
    var (sql, ok) = helper.BeginTransaction();
}

// Get data type by provider
DataType dataType = helper.GetDataType("int");

var (pgDDL, _, _) = pgHelper.GenerateCreateTableSql(entity);

// Different output per provider:
// SQL Server: [Id] INT IDENTITY(1,1)
// PostgreSQL: "Id" SERIAL
