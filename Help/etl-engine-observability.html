Overview

The observability stack provides comprehensive monitoring for pipeline execution:

  • Run logs - Complete execution history with timing and status
  • Data lineage - Column-level tracking of data flow
  • Metrics - Aggregated KPIs with cost modeling
  • Alerts - Rule-based alerting with silence windows
  • Audit - Tamper-evident audit trail
  • Dashboard API - Query interface for dashboards
  • Field redaction - Automatic PII masking in logs

Usage Examples

Setting Up Observability

var store = new ObservabilityStore();
var metrics = new MetricsEngine(store);
var alerting = new AlertingEngine(store, notifierId =>
{
    if (notifierId == "email") return new EmailNotifier();
    if (notifierId == "slack") return new SlackNotifier();
    return null;
});

var dashboard = new PipelineDashboardApi(store, metrics, alerting);

// Attach to pipeline manager
manager.ObservabilityStore = store;
manager.Metrics = metrics;
manager.Alerting = alerting;

Adding Alert Rules

// Alert on failure
alerting.AddRule(new AlertRule
{
    Name = "Pipeline Failure",
    Trigger = AlertTrigger.OnFailure,
    Severity = AlertSeverity.Error,
    SilenceMinutes = 30,
    NotifierIds = new[] { "email", "slack" }
});

// Alert on slow runs
alerting.AddRule(new AlertRule
{
    Name = "Slow Pipeline",
    Trigger = AlertTrigger.OnCondition,
    Condition = "Duration > 300", // 5 minutes
    Severity = AlertSeverity.Warning,
    PipelineIds = new[] { "customer-sync" },
    SilenceMinutes = 60
});

// Alert on missing runs
alerting.AddRule(new AlertRule
{
    Name = "Missing Run",
    Trigger = AlertTrigger.OnNoRunWithin,
    Condition = "HoursWithoutRun > 2",
    Severity = AlertSeverity.Critical,
    PipelineIds = new[] { "critical-pipeline" },
    NotifierIds = new[] { "pagerduty" }
});

Querying Metrics

// Pipeline metrics
var metrics = await dashboard.GetPipelineMetricsAsync(
    "customer-sync",
    from: DateTime.UtcNow.AddDays(-7),
    to: DateTime.UtcNow);

Console.WriteLine($"Success rate: {metrics.SuccessRate:F1}%");
Console.WriteLine($"Avg duration: {metrics.AverageDuration}");
Console.WriteLine($"Records processed: {metrics.TotalRecordsProcessed:N0}");

// Dashboard summary
var summary = await dashboard.GetSummaryAsync(
    DateTime.UtcNow.AddDays(-1),
    DateTime.UtcNow);

Console.WriteLine($"Active runs: {summary.ActiveRuns}");
Console.WriteLine($"Active alerts: {summary.ActiveAlerts}");

Monitoring Active Runs

var active = dashboard.GetActiveRuns();
foreach (var run in active)
{
    Console.WriteLine($"{run.PipelineName}: {run.Progress}% " +
        $"({run.RecordsProcessed:N0} records)");
}

Audit Trail

// Log audit event
await store.AppendAuditAsync(new AuditEntry
{
    Action = "PipelineModified",
    Actor = "admin",
    Resource = "customer-sync",
    Details = "Changed batch size from 100 to 500",
    Timestamp = DateTime.UtcNow
});

// Verify integrity
var tamperedCount = await store.VerifyAuditIntegrityAsync();
if (tamperedCount > 0)
    Console.WriteLine($"WARNING: {tamperedCount} tampered audit entries detected!");
