IDM_Addin

The primary add-in contract. Extends INavigable and IErrorView:

public interface IDM_Addin : INavigable, IErrorView
{
    void Initialize();
    void Dispose();
    void Suspend();
    void Resume();

    AddinDetails Details { get; set; }
    Dependencies Dependencies { get; set; }
    string GetErrorDetails();

    void Run(IPassedArgs pPassedarg);
    void Run(params object[] args);
    Task<IErrorsInfo> RunAsync(IPassedArgs pPassedarg);
    Task<IErrorsInfo> RunAsync(params object[] args);

    string GuidID { get; set; }
    bool IsConfigured { get; set; }
    bool IsRunning { get; set; }
    bool IsSuspended { get; set; }
    bool IsStarted { get; set; }

    void Configure(Dictionary<string, object> settings);
    void ApplyTheme();

    event EventHandler OnStart;
    event EventHandler OnStop;
    event EventHandler<ErrorEventArgs> OnError;
}

public class AddinDetails
{
    public string ParentName { get; set; }
    public string ObjectName { get; set; }
    public string ObjectType { get; set; }
    public string AddinName { get; set; }
    public string Description { get; set; }
    public string GuidID { get; set; }
}

public class Dependencies
{
    public IErrorsInfo ErrorObject { get; set; }
    public IDMLogger Logger { get; set; }
    public IDMEEditor DMEEditor { get; set; }
}

public interface INavigable
{
    void OnNavigatedTo(Dictionary<string, object> parameters);
}

public interface IErrorView
{
    void SetError(string message);
}

Implementation Example

[AddinAttribute(Caption = "Copy Entity Manager", Name = "CopyEntityManager",
    misc = "ImportDataManager", addinType = AddinType.Class)]
public class CopyEntityManager : IDM_Addin
{
    public AddinDetails Details { get; set; }
    public Dependencies Dependencies { get; set; }
    public bool IsConfigured { get; set; }
    public bool IsRunning { get; set; }

    public void Run(IPassedArgs args)
    {
        var ds = Dependencies.DMEEditor.GetDataSource(args.DatasourceName);
        ds.Openconnection();
    }

    public void Initialize() { }
    public void Dispose() { }
    public void Suspend() { }
    public void Resume() { }
    public string GetErrorDetails() => "";
    public void Configure(Dictionary<string, object> settings) { }
    public void ApplyTheme() { }
    public void OnNavigatedTo(Dictionary<string, object> parameters) { }
    public void SetError(string message) { }

    public event EventHandler OnStart;
    public event EventHandler OnStop;
    public event EventHandler<ErrorEventArgs> OnError;
}

Discovery & Registration

Addins are discovered by AssemblyHandler.ScanAssembly() when marked with [AddinAttribute]:

[AddinAttribute(
    Caption = "My Addin",
    Name = "MyAddin",
    addinType = AddinType.Class,
    menu = "Tools",
    iconimage = "my_icon.png")]
public class MyAddin : IDM_Addin { }

// Discovered by AssemblyHandler during LoadAllAssembly()
// Registered in ConfigEditor.Addins list
// Available through DMEEditor.Addins tree

Usage Examples

Accessing Addins

// Find addin by name
var addin = editor.Addins.FirstOrDefault(a => a.Details.AddinName == "MyAddin");
if (addin != null)
{
    addin.Initialize();
    addin.Run(new PassedArgs { Messege = "Execute" });
}

Custom Importer

[AddinAttribute(Caption = "PDF Importer", Name = "PdfImporter",
    addinType = AddinType.Importer)]
public class PdfImporter : IImporter
{
    public string Name { get; set; } = "PDF Importer";
    public string Description { get; set; } = "Import PDF tables";
    public string FileExtension { get; set; } = ".pdf";
    public string MimeType { get; set; } = "application/pdf";

    public bool Import(string path)
    {
        // Extract tables from PDF
        var tables = ExtractTables(path);
        return tables.Count > 0;
    }
}
