Architecture Overview

The RDBMSHelpers subsystem answers the question: "Given a database type and a table/entity structure, generate the correct SQL for every operation — schema queries, DDL, DML, feature checks, and entity validation."

Originally a single monolithic class, it was refactored into a layered architecture with 6 top-level helpers and 18 sub-files in DMLHelpers/ and EntityHelpers/. Every method accepts a DataSourceType enum and produce database-specific SQL.

DatabaseSchemaQueryHelper — Metadata Queries

Generates database-specific SQL for inspecting schema metadata across 25+ database types.

Usage

// List all schemas/databases accessible to a user
string sql = RDBMSHelper.GetSchemasorDatabases(DataSourceType.SqlServer, null);
// → "SELECT name FROM sys.databases"

string pg = RDBMSHelper.GetSchemasorDatabases(DataSourceType.Postgre, "myuser");
// → "SELECT schema_name FROM information_schema.schemata
//     WHERE schema_owner = 'myuser'"

// Safe version with error handling
var (query, success, error) = RDBMSHelper.GetSchemasorDatabasesSafe(
    DataSourceType.Oracle, "SCOTT", throwOnError: false);

// Check if a table exists
string check = RDBMSHelper.GetTableExistsQuery(DataSourceType.MySQL, "products", "inventory");
// → "SELECT COUNT(*) FROM information_schema.tables
//     WHERE table_schema = 'inventory' AND table_name = 'products'"

// Get column metadata
string cols = RDBMSHelper.GetColumnInfoQuery(DataSourceType.Postgre, "products", "public");
// → PostgreSQL information_schema.columns query
Supported database types: SQL Server, MySQL, PostgreSQL, Oracle, DB2, SQLite, Firebird, MongoDB, Cassandra, Redis, Elasticsearch, Snowflake, BigQuery, Redshift, Vertica, ClickHouse, Couchbase, Neo4j, and more.

DatabaseObjectCreationHelper — DDL Generation

Generates CREATE TABLE, indexes, primary keys, DROP, and TRUNCATE statements from EntityStructure objects. Uses DataTypeFieldMappingHelper for column type resolution.

Usage

// Create a table from an entity definition
var entity = new EntityStructure
{
    DataSourceType = DataSourceType.SqlServer,
    EntityName = "Products",
    Fields = new List<EntityField>
    {
        new() { FieldName = "Id", Fieldtype = "System.Int32", IsKey = true },
        new() { FieldName = "Name", Fieldtype = "System.String", Size1 = 200 },
        new() { FieldName = "Price", Fieldtype = "System.Decimal",
                NumericPrecision = 18, NumericScale = 2 }
    }
};

var (sql, success, error) = RDBMSHelper.GenerateCreateTableSQL(entity);
// → "CREATE TABLE Products (
//       Id INT IDENTITY(1,1) NOT NULL,
//       Name NVARCHAR(200) NULL,
//       Price DECIMAL(18,2) NULL,
//       PRIMARY KEY (Id))"

// Add indexes
string idx = RDBMSHelper.GenerateCreateIndexQuery(
    DataSourceType.Postgre, "Products", "idx_products_name",
    new[] { "Name" }, new Dictionary<string,object> { ["Unique"] = false });

// Drop and truncate
string drop = RDBMSHelper.GetDropEntity(DataSourceType.MySQL, "products");
// → "DROP TABLE IF EXISTS products"
string trunc = RDBMSHelper.GetTruncateTableQuery(DataSourceType.SqlServer, "products", "dbo");
// → "TRUNCATE TABLE dbo.products"

DatabaseFeatureHelper — Capability Detection

Answers: "Does PostgreSQL support JSON columns?" or "Should I use sequences or IDENTITY on this database?"

7 feature categories

public enum DatabaseFeature
{
    WindowFunctions,   // ROW_NUMBER(), RANK(), etc.
    Json,              // Native JSON type & operations
    Xml,               // Native XML type & operations
    TemporalTables,    // System-versioned temporal tables
    FullTextSearch,    // Full-text indexing & search
    Partitioning,      // Table partitioning
    ColumnStore        // Columnar storage format
}

Usage

// Runtime feature check — avoids generating invalid SQL
if (RDBMSHelper.SupportsFeature(DataSourceType.MySQL, DatabaseFeature.Json))
{
    // Safe to generate JSON column type
}

// Get all features for a database
var features = RDBMSHelper.GetSupportedFeatures(DataSourceType.Oracle);
// → [WindowFunctions, Json, Xml, Partitioning, ...]

// Sequence operations
string seqSql = RDBMSHelper.GenerateFetchNextSequenceValueQuery(
    DataSourceType.Postgre, "order_id_seq");
// → "SELECT nextval('order_id_seq')"

// Last identity value (after INSERT)
string idSql = RDBMSHelper.GenerateFetchLastIdentityQuery(DataSourceType.SqlServer);
// → "SELECT SCOPE_IDENTITY()"

// Transaction statements
string begin = RDBMSHelper.GetTransactionStatement(
    DataSourceType.Postgre, TransactionOperation.Begin);
// → "BEGIN TRANSACTION"

// Identifier limits
int maxLen = RDBMSHelper.GetMaxIdentifierLength(DataSourceType.Oracle);
// → 30  (Oracle max identifier length)

if (RDBMSHelper.SupportsAutoIncrement(DataSourceType.SqlLite))
    // Use AUTOINCREMENT syntax

DatabaseQueryRepositoryHelper — 200+ Predefined Queries

Seeds the engine with schema-inspection queries for 30+ database types. Called by EnvironmentService.AddAllDataSourceQueryConfigurations() during startup.

Query types

// Sqlcommandtype enum values used to look up queries:
//   getTable                    → SELECT * FROM {table} (or equivalent)
//   getlistoftables             → List all tables/collections
//   getlistoftablesfromotherschema → Cross-schema table listing
//   getPKforTable               → Primary key column identification
//   getFKforTable               → Foreign key relationships
//   getChildTable               → Child table discovery
//   getParentTable              → Parent table discovery

Usage

// Get a predefined query (cached)
string sql = RDBMSHelper.GetQuery(DataSourceType.Postgre, Sqlcommandtype.getPKforTable);
// → PostgreSQL query to find primary key columns

// Get ALL queries for a specific database type
var queries = DatabaseQueryRepositoryHelper.GetQueriesForDatabase(DataSourceType.SqlServer);
// → Dictionary mapping Sqlcommandtype → SQL string

// Validate SQL before executing
bool ok = RDBMSHelper.IsSqlStatementValid("SELECT * FROM users");
// → true (contains SQL keywords)

// Get repository statistics
var stats = DatabaseQueryRepositoryHelper.GetQueryStatistics();
// → { TotalQueries: 200+, Databases: 30+, Categories: 8 }

// Create the repository programmatically (called at startup)
var repos = RDBMSHelper.CreateQuerySqlRepos();
// → List of 200+ QuerySqlRepo objects for all supported DBs

Entity Helpers — Entity-Driven SQL Generation

Rather than passing raw table names and dictionaries, these helpers generate SQL directly from EntityStructure objects, validating along the way.

Entity validation — 15+ rules

var (isValid, errors) = RDBMSHelper.ValidateEntityStructure(entity);
if (!isValid)
{
    foreach (var err in errors)
        Console.WriteLine(err);
    // "Entity name 'Select' is a reserved keyword"
    // "Field '123abc' contains invalid characters"
    // "No primary key defined for entity 'Users'"
    // "Field 'email' is a duplicate"
}

// Get compatibility info across database types
var compat = RDBMSHelper.GetEntityCompatibilityInfo(entity);
// → { SqlServer: "compatible", Oracle: "compatible",
//     MySQL: "column 'Description' exceeds max length" }

// Get improvement suggestions
var suggestions = RDBMSHelper.SuggestEntityImprovements(entity);
// → ["Consider adding a CreatedAt timestamp field",
//     "Field 'User_name' should use PascalCase: 'UserName'",
//     "Index recommended for foreign key field 'CategoryId'"]

Entity-based CRUD

// INSERT from entity definition
var (sql, parameters, success, error) = RDBMSHelper.GenerateInsertWithValues(
    entity,
    new Dictionary<string,object> {
        ["Name"] = "Widget",
        ["Price"] = 19.99m
    });
// → ("INSERT INTO Products (Name, Price) VALUES (@Name, @Price)",
//     { Name: "Widget", Price: 19.99 }, true, null)

// UPDATE with conditions
var (sql, success, error) = RDBMSHelper.GenerateUpdateEntityWithValues(
    entity,
    new Dictionary<string,object> { ["Price"] = 24.99m },   // SET
    new Dictionary<string,object> { ["Id"] = 42 });         // WHERE
// → "UPDATE Products SET Price = 24.99 WHERE Id = 42"

// DELETE
var (sql, success, error) = RDBMSHelper.GenerateDeleteEntityWithValues(
    entity,
    new Dictionary<string,object> { ["Id"] = 42 });
// → "DELETE FROM Products WHERE Id = 42"

End-to-End Usage Scenarios

A: Create a table from a C# model

var entity = new EntityStructure
{
    DataSourceType = DataSourceType.Postgre,
    EntityName = "Orders",
    SchemaName = "public",
    Fields = new List<EntityField>
    {
        new() { FieldName = "Id", Fieldtype = "System.Int64", IsKey = true },
        new() { FieldName = "CustomerId", Fieldtype = "System.Int32", IsKey = false },
        new() { FieldName = "OrderDate", Fieldtype = "System.DateTime" },
        new() { FieldName = "Total", Fieldtype = "System.Decimal",
                NumericPrecision = 12, NumericScale = 2 }
    }
};

var (ddl, ok, err) = RDBMSHelper.GenerateCreateTableSQL(entity);
// DDL: CREATE TABLE public.Orders (
//   Id BIGSERIAL NOT NULL,
//   CustomerId INT NULL,
//   OrderDate TIMESTAMP NULL,
//   Total NUMERIC(12,2) NULL,
//   PRIMARY KEY (Id))

// Execute via IDataSource
ds.ExecuteSql(ddl);

// Add an index on CustomerId
string idx = RDBMSHelper.GenerateCreateIndexQuery(
    DataSourceType.Postgre, "Orders", "idx_orders_customer",
    new[] { "CustomerId" });
ds.ExecuteSql(idx);

B: Schema migration — check before CREATE

string checkSql = RDBMSHelper.GetTableExistsQuery(
    DataSourceType.SqlServer, "Products", "dbo");
// → "SELECT COUNT(*) FROM INFORMATION_SCHEMA.TABLES
//     WHERE TABLE_SCHEMA = 'dbo' AND TABLE_NAME = 'Products'"

var result = ds.RunQuery(checkSql);
if (result.Rows[0][0] == 0)
{
    var (ddl, ok, err) = RDBMSHelper.GenerateCreateTableSQL(entity);
    ds.ExecuteSql(ddl);
}

C: Cross-database pagination

string BuildPagedQuery(DataSourceType dbType, string baseQuery, int page, int size)
{
    string paging = RDBMSHelper.GetPagingSyntax(dbType, page, size);
    return $"{baseQuery} {paging}";
}

// SQL Server: SELECT * FROM Products ORDER BY Id OFFSET 10 ROWS FETCH NEXT 10 ROWS ONLY
// MySQL:      SELECT * FROM Products ORDER BY Id LIMIT 10 OFFSET 10
// Oracle:     SELECT * FROM Products ORDER BY Id OFFSET 10 ROWS FETCH NEXT 10 ROWS ONLY
// Firebird:   SELECT * FROM Products ORDER BY Id ROWS 11 TO 20
