Overview

The Beep Data Management Engine is a comprehensive, multi-target framework that provides powerful data access patterns, unit of work implementations, change tracking, and enterprise-level data management capabilities for modern .NET applications.

Built with extensibility in mind, the engine supports a wide variety of data sources from traditional relational databases to modern vector databases, file formats, and custom data providers through its plugin-based architecture.

Basic Usage

Here's a simple example to get you started:

Creating Your First UnitOfWork

using Autofac;
using TheTechIdea.Beep;
using TheTechIdea.Beep.ConfigUtil;
using TheTechIdea.Beep.Logger;
using TheTechIdea.Util;

// Initialize DME Editor using Autofac
var builder = new ContainerBuilder();
builder.RegisterType<DMEEditor>().As<IDMEEditor>().SingleInstance();
builder.RegisterType<ConfigEditor>().As<IConfigEditor>().SingleInstance();
builder.RegisterType<DMLogger>().As<IDMLogger>().SingleInstance();
builder.RegisterType<Util>().As<IUtil>().SingleInstance();
builder.RegisterType<ErrorsInfo>().As<IErrorsInfo>().SingleInstance();
builder.RegisterType<JsonLoader>().As<IJsonLoader>().SingleInstance();
builder.RegisterType<AssemblyHandler>().As<IAssemblyHandler>().SingleInstance();

var container = builder.Build();
var dmeEditor = container.Resolve<IDMEEditor>();

// Initialize required configurations for IDataSource support
InitializeBeepDefaults(dmeEditor);

// Create a data connection
ConnectionProperties connectionProps = new ConnectionProperties
{
    ConnectionName = "MyDatabase",
    DatabaseType = DataSourceType.SqlServer,
    ConnectionString = "Server=localhost;Database=MyDB;Integrated Security=true;"
};

// Add connection to configuration
dmeEditor.ConfigEditor.AddDataConnection(connectionProps);

// Get data source and open connection
IDataSource dataSource = dmeEditor.GetDataSource("MyDatabase");
dataSource.Openconnection();

// Create UnitOfWork for your entity
var unitOfWork = new UnitofWork<Customer>(
    dmeEditor, 
    "MyDatabase", 
    "Customers", 
    "CustomerId"
);

// Load data
await unitOfWork.Get();

// Work with your data
unitOfWork.New(); // Create new customer
var newCustomer = unitOfWork.CurrentItem;
newCustomer.Name = "John Doe";
newCustomer.Email = "john@example.com";

// Save changes
await unitOfWork.Commit();
