üèóÔ∏è Overview & Core Pattern

The RegisterBeep class sets up the Beep Data Management Engine for direct data operations. Once configured, you work with DMEEditor to manage data sources, create connections, and perform data operations across multiple database types.

üéØ Purpose

RegisterBeep provides the foundation for data management - after setup, you use dmeEditor directly for all data operations.

1Ô∏è‚É£ Bootstrap

Register services with Autofac and get IDMEEditor

2Ô∏è‚É£ Initialize

Add default configurations for drivers, mappings, and queries

3Ô∏è‚É£ Connect

Create connection configurations for your data sources

4Ô∏è‚É£ Operate

Perform CRUD operations and data management tasks

‚öôÔ∏è Initial Setup & Registration

‚ö†Ô∏è CRITICAL SEQUENCE

The initialization must follow this exact order:

  1. Initialize DMEEditor with Autofac dependency injection
  2. Load ConnectionHelper drivers using GetAllConnectionConfigs() - THIS IS MANDATORY
  3. Configure ConfigEditor with specialized managers
  4. Create specific connections for your application

Step 1: Basic Autofac Setup

using Autofac;
using TheTechIdea.Beep;
using TheTechIdea.Beep.ConfigUtil;
using TheTechIdea.Beep.Logger;
using TheTechIdea.Util;
using TheTechIdea.Beep.Helpers;

static void Main()
{
    // Create Autofac container builder
    var builder = new ContainerBuilder();
    
    // Register BeepDM core services
    builder.RegisterType<DMEEditor>().As<IDMEEditor>().SingleInstance();
    builder.RegisterType<ConfigEditor>().As<IConfigEditor>().SingleInstance();
    builder.RegisterType<DMLogger>().As<IDMLogger>().SingleInstance();
    builder.RegisterType<Util>().As<IUtil>().SingleInstance();
    builder.RegisterType<ErrorsInfo>().As<IErrorsInfo>().SingleInstance();
    builder.RegisterType<JsonLoader>().As<IJsonLoader>().SingleInstance();
    builder.RegisterType<AssemblyHandler>().As<IAssemblyHandler>().SingleInstance();
    
    // Build container and resolve DMEEditor
    var container = builder.Build();
    var dmeEditor = container.Resolve<IDMEEditor>();
    
    Console.WriteLine("‚úÖ BeepDM core services initialized successfully");
    
    // STEP 2: CRITICAL - Initialize connection drivers FIRST
    InitializeConnectionDrivers(dmeEditor);
    
    // STEP 3: Configure ConfigEditor (now that drivers are loaded)
    ConfigureConfigEditor(dmeEditor);
    
    // STEP 4: Create your application connections
    CreateApplicationConnections(dmeEditor);
}

Step 2: ConnectionHelper Driver Initialization (CRITICAL)

/// <summary>
/// STEP 2: Initialize all connection drivers from ConnectionHelper
/// This MUST be done BEFORE creating any connections
/// </summary>
public static void InitializeConnectionDrivers(IDMEEditor dmeEditor)
{
    try
    {
        Console.WriteLine("üîß Initializing connection drivers from ConnectionHelper...");
        
        // Get all 39 predefined connection configurations from ConnectionHelper
        var allConnectionConfigs = ConnectionHelper.GetAllConnectionConfigs();
        
        Console.WriteLine($"üì¶ Loading {allConnectionConfigs.Count} connection driver configurations:");
        
        // Add each configuration to the DMEEditor
        foreach (var config in allConnectionConfigs)
        {
            // Avoid duplicates by checking GuidID
            if (!dmeEditor.ConfigEditor.DataDriversClasses.Any(d => d.GuidID == config.GuidID))
            {
                dmeEditor.ConfigEditor.DataDriversClasses.Add(config);
            }
        }
        
        // Save configurations to persistent storage
        dmeEditor.ConfigEditor.SaveConnectionDriversConfigValues();
        
        Console.WriteLine($"‚úÖ Successfully loaded {allConnectionConfigs.Count} driver configurations");
        
        // Show what was loaded by category
        var driversByCategory = allConnectionConfigs.GroupBy(c => c.DatasourceCategory);
        foreach (var categoryGroup in driversByCategory)
        {
            Console.WriteLine($"   üìÅ {categoryGroup.Key}: {categoryGroup.Count()} drivers");
        }
        
        Console.WriteLine("üìä Available driver categories:");
        Console.WriteLine("   üè¢ RDBMS: Enterprise databases (SQL Server, Oracle, PostgreSQL, MySQL, etc.)");
        Console.WriteLine("   üìÑ FILE: File formats (CSV, Excel, JSON, Text)");
        Console.WriteLine("   üóÑÔ∏è NoSQL: Document & NoSQL databases (MongoDB, CouchDB, etc.)");
        Console.WriteLine("   üß† VectorDB: AI/ML vector databases (ChromaDB, Pinecone, Qdrant, etc.)");
        Console.WriteLine("   üåê WebAPI: REST and web service connections");
        Console.WriteLine("   ‚òÅÔ∏è Cloud: Cloud and streaming services");
        
        // Verify critical drivers are available
        VerifyEssentialDrivers(dmeEditor);
    }
    catch (Exception ex)
    {
        Console.WriteLine($"‚ùå CRITICAL ERROR: Failed to initialize connection drivers: {ex.Message}");
        Console.WriteLine("üí° Cannot proceed without connection drivers. Application will not function properly.");
        throw;
    }
}

/// <summary>
/// Verify that essential drivers are available for basic operations
/// </summary>
private static void VerifyEssentialDrivers(IDMEEditor dmeEditor)
{
    var essentialDrivers = new[]
    {
        DataSourceType.SqlLite,    // Most commonly used embedded database
        DataSourceType.CSV,        // File operations
        DataSourceType.SqlServer,  // Enterprise database
        DataSourceType.Xls         // Excel files
    };
    
    Console.WriteLine("üîç Verifying essential drivers...");
    
    foreach (var driverType in essentialDrivers)
    {
        var driver = dmeEditor.ConfigEditor.DataDriversClasses
            .FirstOrDefault(d => d.DatasourceType == driverType);
        
        if (driver != null)
        {
            Console.WriteLine($"   ‚úÖ {driverType}: {driver.PackageName} v{driver.version}");
        }
        else
        {
            Console.WriteLine($"   ‚ùå {driverType}: NOT FOUND");
        }
    }
}

/// <summary>
/// STEP 3: Configure ConfigEditor after connection drivers are loaded
/// This step sets up specialized managers and persistent configuration
/// </summary>
public static void ConfigureConfigEditor(IDMEEditor dmeEditor)
{
    try
    {
        Console.WriteLine("‚öôÔ∏è Configuring ConfigEditor with specialized managers...");
        
        var configEditor = dmeEditor.ConfigEditor;
        
        // Verify that drivers were loaded in Step 2
        if (configEditor.DataDriversClasses.Count == 0)
        {
            throw new InvalidOperationException("No connection drivers found. Did you run InitializeConnectionDrivers() first?");
        }
        
        Console.WriteLine($"‚úÖ Found {configEditor.DataDriversClasses.Count} connection drivers loaded");
        
        // Set configuration type and ensure proper paths
        configEditor.ConfigType = BeepConfigType.DataConnections;
        
        // Set up configuration path if not already set
        if (string.IsNullOrEmpty(configEditor.ConfigPath))
        {
            configEditor.ConfigPath = Path.Combine(
                Environment.GetFolderPath(Environment.SpecialFolder.ApplicationData), 
                "BeepDataEngine", 
                "Config"
            );
        }
        
        // Ensure configuration directory exists
        Directory.CreateDirectory(configEditor.ConfigPath);
        Console.WriteLine($"üìÅ Config directory: {configEditor.ConfigPath}");
        
        // Initialize configuration collections if they're null
        InitializeConfigCollections(configEditor);
        
        // Load existing configuration values from persistent storage
        configEditor.LoadConfigValues();
        
        // Add default queries and mappings
        AddDefaultConfigurations(configEditor);
        
        Console.WriteLine("‚úÖ ConfigEditor configuration completed");
        ShowConfigSummary(configEditor);
    }
    catch (Exception ex)
    {
        Console.WriteLine($"‚ùå Failed to configure ConfigEditor: {ex.Message}");
        throw;
    }
}

/// <summary>
/// Initialize configuration collections to ensure they're not null
/// </summary>
private static void InitializeConfigCollections(IConfigEditor configEditor)
{
    configEditor.DataConnections ??= new List<ConnectionProperties>();
    configEditor.DataSourcesClasses ??= new List<AssemblyClassDefinition>();
    // DataDriversClasses should already be populated from Step 2
    configEditor.EntityMaps ??= new List<EntityDataMap>();
    configEditor.DataTypesMap ??= new List<DataTypesMap>();
    configEditor.QueryList ??= new List<QuerySqlRepo>();
    configEditor.WorkFlowStepsConfig ??= new List<WorkFlowStepsConfig>();
    
    Console.WriteLine("üìã Configuration collections initialized");
}

/// <summary>
/// Add default configurations like common queries and mappings
/// </summary>
private static void AddDefaultConfigurations(IConfigEditor configEditor)
{
    // Add common SQL queries if none exist
    if (configEditor.QueryList.Count == 0)
    {
        AddCommonQueries(configEditor);
    }
    
    // Add basic entity mappings if none exist
    if (configEditor.EntityMaps.Count == 0)
    {
        AddBasicEntityMappings(configEditor);
    }
    
    Console.WriteLine("üîß Default configurations added");
}

SQLite Database Connection

public static bool CreateConnectionForSqlite(IDMEEditor dmeEditor)
{
    try
    {
        // Get the connection driver configuration
        var config = dmeEditor.ConfigEditor.DataDriversClasses
            .FirstOrDefault(p => p.DatasourceType == DataSourceType.SqlLite);
        
        if (config == null)
        {
            Console.WriteLine("‚ùå SQLite driver not found");
            return false;
        }
        
        // Create connection properties
        var connectionProperties = new ConnectionProperties
        {
            ConnectionString = "Data Source=./Beep/dbfiles/northwind.db",
            ConnectionName = "northwind.db",
            DriverName = config.PackageName,
            DriverVersion = config.version,
            DatabaseType = DataSourceType.SqlLite,
            Category = DatasourceCategory.RDBMS
        };
        
        // Add to connection list
        dmeEditor.ConfigEditor.AddDataConnection(connectionProperties);
        
        // Get and test the data source
        var sqliteDataSource = dmeEditor.GetDataSource("northwind.db");
        if (sqliteDataSource != null)
        {
            sqliteDataSource.Openconnection();
            
            if (sqliteDataSource.ConnectionStatus == ConnectionState.Open)
            {
                Console.WriteLine("‚úÖ SQLite connection established");
                return true;
            }
        }
        
        Console.WriteLine("‚ùå Failed to connect to SQLite database");
        return false;
    }
    catch (Exception ex)
    {
        Console.WriteLine($"‚ùå SQLite connection error: {ex.Message}");
        return false;
    }
}

Generic Connection Creator

public static bool CreateConnection(IDMEEditor dmeEditor, string connectionName, DataSourceType dataSourceType, 
    string connectionString = null, string filePath = null, string fileName = null)
{
    try
    {
        // Get driver configuration
        var config = dmeEditor.ConfigEditor.DataDriversClasses
            .FirstOrDefault(p => p.DatasourceType == dataSourceType);
        
        if (config == null)
        {
            Console.WriteLine($"‚ùå Driver for {dataSourceType} not found");
            return false;
        }
        
        // Build connection properties based on data source type
        var connectionProperties = new ConnectionProperties
        {
            ConnectionName = connectionName,
            DriverName = config.PackageName,
            DriverVersion = config.version,
            DatabaseType = dataSourceType,
            Category = GetDataSourceCategory(dataSourceType)
        };
        
        // Set type-specific properties
        if (dataSourceType == DataSourceType.SqlLite || 
            dataSourceType == DataSourceType.SqlServer ||
            dataSourceType == DataSourceType.MySQL ||
            dataSourceType == DataSourceType.Postgre)
        {
            connectionProperties.ConnectionString = connectionString;
        }
        else if (IsFileBasedDataSource(dataSourceType))
        {
            connectionProperties.FilePath = filePath;
            connectionProperties.FileName = fileName;
            connectionProperties.Ext = Path.GetExtension(fileName)?.TrimStart('.');
        }
        
        // Add connection and test
        dmeEditor.ConfigEditor.AddDataConnection(connectionProperties);
        
        var dataSource = dmeEditor.GetDataSource(connectionName);
        if (dataSource != null)
        {
            dataSource.Openconnection();
            
            if (dataSource.ConnectionStatus == ConnectionState.Open)
            {
                Console.WriteLine($"‚úÖ {dataSourceType} connection '{connectionName}' established");
                return true;
            }
        }
        
        Console.WriteLine($"‚ùå Failed to connect to {dataSourceType} '{connectionName}'");
        return false;
    }
    catch (Exception ex)
    {
        Console.WriteLine($"‚ùå Connection error for {connectionName}: {ex.Message}");
        return false;
    }
}

private static DatasourceCategory GetDataSourceCategory(DataSourceType dataSourceType)
{
    return dataSourceType switch
    {
        DataSourceType.SqlServer or DataSourceType.MySQL or 
        DataSourceType.Postgre or DataSourceType.SqlLite => DatasourceCategory.RDBMS,
        
        DataSourceType.CSV or DataSourceType.Xls or 
        DataSourceType.Text => DatasourceCategory.FILE,
        
        DataSourceType.MongoDB => DatasourceCategory.NOSQL,
        
        DataSourceType.WebAPI or DataSourceType.REST => DatasourceCategory.WEBAPI,
        
        _ => DatasourceCategory.RDBMS
    };
}

private static bool IsFileBasedDataSource(DataSourceType dataSourceType)
{
    return dataSourceType is DataSourceType.CSV or DataSourceType.Xls or 
           DataSourceType.Text or DataSourceType.Json;
}

Basic Data Access Pattern

// Core pattern for all data operations
public static void WorkWithDataSource(IDMEEditor dmeEditor, string connectionName)
{
    try
    {
        // Get data source from DMEEditor
        var dataSource = dmeEditor.GetDataSource(connectionName);
        
        if (dataSource == null)
        {
            Console.WriteLine($"‚ùå Data source '{connectionName}' not found");
            return;
        }
        
        // Ensure connection is open
        dataSource.Openconnection();
        
        if (dataSource.ConnectionStatus != ConnectionState.Open)
        {
            Console.WriteLine($"‚ùå Failed to open connection to '{connectionName}'");
            return;
        }
        
        Console.WriteLine($"‚úÖ Connected to '{connectionName}'");
        
        // Get list of entities (tables/sheets/collections)
        var entities = dataSource.GetEntitesList();
        Console.WriteLine($"üìã Found {entities.Count} entities:");
        
        foreach (var entity in entities)
        {
            Console.WriteLine($"   ‚Ä¢ {entity}");
        }
        
        // Work with first entity if available
        if (entities.Count > 0)
        {
            string firstEntity = entities[0];
            
            // Get entity structure (schema information)
            var structure = dataSource.GetEntityStructure(firstEntity, true);
            Console.WriteLine($"üìä Entity '{firstEntity}' has {structure.Fields?.Count ?? 0} fields");
            
            // Get data from entity
            var data = dataSource.GetEntity(firstEntity, null);
            
            if (data != null)
            {
                var dataList = data as IList ?? new List<object> { data };
                Console.WriteLine($"üìà Retrieved {dataList.Count} records from '{firstEntity}'");
                
                // Process the data
                ProcessEntityData(firstEntity, data, structure);
            }
        }
    }
    catch (Exception ex)
    {
        Console.WriteLine($"‚ùå Error working with data source: {ex.Message}");
        dmeEditor.AddLogMessage("Beep", 
            $"Data operation error: {ex.Message}", 
            DateTime.Now, -1, null, Errors.Failed);
    }
    finally
    {
        // Always clean up
        dataSource?.Closeconnection();
    }
}

// Real implementation from BeepSharedFunctions
public static IErrorsInfo MoveEntity(IDMEEditor dmeEditor, string sourceName, string destName, 
    string entityName, Progress<PassedArgs> progress = null)
{
    IDataSource sourceDataSource = null;
    IDataSource destDataSource = null;
    
    try
    {
        // Get source and destination data sources
        sourceDataSource = dmeEditor.GetDataSource(sourceName);
        destDataSource = dmeEditor.GetDataSource(destName);
        
        if (sourceDataSource == null || destDataSource == null)
        {
            var errorMsg = $"Source '{sourceName}' or destination '{destName}' not found";
            dmeEditor.AddLogMessage("Beep", errorMsg, 
                DateTime.Now, 0, "", Errors.Failed);
            return dmeEditor.ErrorObject;
        }
        
        // Open connections
        sourceDataSource.Openconnection();
        destDataSource.Openconnection();
        
        if (sourceDataSource.ConnectionStatus != ConnectionState.Open)
        {
            throw new InvalidOperationException($"Cannot open source connection to '{sourceName}'");
        }
        
        if (destDataSource.ConnectionStatus != ConnectionState.Open)
        {
            throw new InvalidOperationException($"Cannot open destination connection to '{destName}'");
        }
        
        Console.WriteLine($"üîÑ Moving entity '{entityName}' from {sourceName} to {destName}");
        
        // Get entity structure from source
        var srcEntityStructure = sourceDataSource.GetEntityStructure(entityName, true);
        if (srcEntityStructure == null)
        {
            throw new InvalidOperationException($"Entity '{entityName}' not found in source '{sourceName}'");
        }
        
        // Get data from source
        var sourceData = sourceDataSource.GetEntity(entityName, null);
        if (sourceData == null)
        {
            Console.WriteLine($"‚ö†Ô∏è No data found in source entity '{entityName}'");
            return dmeEditor.ErrorObject;
        }
        
        // Create entity in destination if it doesn't exist
        if (!destDataSource.CheckEntityExist(entityName))
        {
            Console.WriteLine($"üìã Creating entity '{entityName}' in destination");
            var createResult = destDataSource.CreateEntityAs(srcEntityStructure);
            
            if (createResult.Flag != Errors.Ok)
            {
                throw new InvalidOperationException($"Failed to create entity in destination: {createResult.Message}");
            }
        }
        
        // Move the data with progress reporting
        var updateResult = destDataSource.UpdateEntities(entityName, sourceData, progress);
        
        if (updateResult.Flag == Errors.Ok)
        {
            var dataList = sourceData as IList ?? new List<object> { sourceData };
            Console.WriteLine($"‚úÖ Successfully moved {dataList.Count} records");
        }
        else
        {
            Console.WriteLine($"‚ùå Failed to move data: {updateResult.Message}");
        }
        
        return updateResult;
    }
    catch (Exception ex)
    {
        var errorMsg = $"Error moving entity '{entityName}': {ex.Message}";
        Console.WriteLine($"‚ùå {errorMsg}");
        
        dmeEditor.AddLogMessage("Beep", errorMsg, 
            DateTime.Now, -1, ex.StackTrace, Errors.Failed);
        
        return dmeEditor.ErrorObject;
    }
    finally
    {
        // Clean up connections
        sourceDataSource?.Closeconnection();
        destDataSource?.Closeconnection();
    }
}
