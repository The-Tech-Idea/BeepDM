DLL Loading and Full Scan Flow

// Typical startup flow
IAssemblyHandler assemblyHandler = dmeEditor.assemblyHandler;

// Full discovery: runtime + folder loads + scans + extension passes
await Task.Run(() => assemblyHandler.LoadAllAssembly(progress, cancellationToken));

// Optional: targeted load from a specific DLL path
Assembly pluginAssembly = assemblyHandler.LoadAssembly(@"C:\Plugins\MySourcePlugin.dll");

// Optional: load an entire folder and scan for datasource classes
var folderAssemblies = assemblyHandler.LoadAssembliesFromFolder(
    @"C:\Plugins",
    FolderFileTypes.DataSources,
    scanForDataSources: true);
Load sequence note

LoadAllAssembly() currently orchestrates builtin/runtime loading, extension discovery, folder loading, driver/datasource scanning, and extension scans in one pass.

NuGet Management Flow

// Search packages
var packages = await assemblyHandler.SearchNuGetPackagesAsync("postgres", take: 10);

// Inspect versions
var versions = await assemblyHandler.GetNuGetPackageVersionsAsync("Npgsql");

// Configure custom source if needed
assemblyHandler.AddNuGetSource("my-feed", "https://my-feed/v3/index.json", isEnabled: true);
var activeSources = assemblyHandler.GetActiveSourceUrls();

// Download + load package and dependencies
var loaded = await assemblyHandler.LoadNuggetFromNuGetAsync(
    packageName: "MyCompany.CustomDataSource",
    version: versions.FirstOrDefault(),
    sources: activeSources);

// Driver provenance lookup
bool fromNuGet = assemblyHandler.IsDriverFromNuGet("MyCompany.CustomDataSource.CustomDriver");
var mappings = assemblyHandler.GetAllDriverPackageMappings();

Create an ILoaderExtention for New Data Classes

using System.Reflection;
using TheTechIdea.Beep.Addin;
using TheTechIdea.Beep.ConfigUtil;
using TheTechIdea.Beep.Tools;
using TheTechIdea.Beep.Utilities;

public sealed class CustomLoaderExtension : ILoaderExtention
{
    private readonly AssemblyHandler _handler;
    private readonly IConfigEditor _config;

    public CustomLoaderExtension(AssemblyHandler handler)
    {
        _handler = handler;
        _config = handler.ConfigEditor;
    }

    public IErrorsInfo LoadAllAssembly()
    {
        // Optional: extension-specific bootstrap
        return _handler.ErrorObject;
    }

    public IErrorsInfo Scan(assemblies_rep assembly)
    {
        if (assembly?.DllLib != null) return Scan(assembly.DllLib);
        return _handler.ErrorObject;
    }

    public IErrorsInfo Scan(Assembly assembly)
    {
        foreach (var type in assembly.GetTypes())
        {
            var ti = type.GetTypeInfo();
            if (ti.ImplementedInterfaces.Contains(typeof(IDataSource)))
            {
                var cls = _handler.GetAssemblyClassDefinition(ti, "IDataSource");
                _config.DataSourcesClasses.Add(cls);
            }
        }
        return _handler.ErrorObject;
    }

    public IErrorsInfo Scan()
    {
        foreach (var asm in _handler.LoadedAssemblies) _ = Scan(asm);
        return _handler.ErrorObject;
    }
}
Contract name is ILoaderExtention

The current interface spelling in the codebase is ILoaderExtention (not ILoaderExtension).
