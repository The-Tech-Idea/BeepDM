## Objective

Create the foundational `ProxyCluster` class and its supporting types.  
After this phase, a caller can construct a cluster of N proxy nodes and route any
`IDataSource` call through the cluster exactly as if talking to a single datasource.

## 1. `IProxyNode` Interface

```csharp
namespace TheTechIdea.Beep.Proxy
{
    /// <summary>
    /// A single addressable proxy node within a <see cref="ProxyCluster"/>.
    /// Wraps an <see cref="IProxyDataSource"/> with cluster-level metadata.
    /// </summary>
    public interface IProxyNode
    {
        string             NodeId       { get; }
        IProxyDataSource   Proxy        { get; }

        /// <summary>Relative routing weight (1 = normal, 2 = twice as likely to be picked).</summary>
        int                Weight       { get; set; }

        /// <summary>Cluster-tier role: Primary receives writes; Replica receives reads.</summary>
        ProxyDataSourceRole NodeRole    { get; set; }

        /// <summary>Last liveness probe result.</summary>
        bool               IsAlive      { get; }

        /// <summary>UTC timestamp of the last probe attempt.</summary>
        DateTime           LastProbeUtc { get; }

        /// <summary>Number of in-flight operations currently routed to this node.</summary>
        int                InFlightCount { get; }

        /// <summary>Node-scoped aggregated metrics (populated from node.Proxy.GetMetrics()).</summary>
        DataSourceMetrics  Metrics      { get; }

        /// <summary>UTC timestamp when this node was added to the cluster (used for slow-start).</summary>
        DateTime           AddedAtUtc   { get; }
    }
}
```

```csharp
namespace TheTechIdea.Beep.Proxy
{
    /// <summary>
    /// Extends <see cref="IProxyDataSource"/> with cluster-tier node management,
    /// policy fan-out, and aggregate observability.
    /// Callers who only need <see cref="IDataSource"/> never need to cast to this.

        event EventHandler<RoleChangeEventArgs>          OnNodePromoted;
```

## 3. `ProxyNode` (concrete)

**File:** `ProxyNode.cs`

```csharp
namespace TheTechIdea.Beep.Proxy
{
    public sealed class ProxyNode : IProxyNode
    {
        private int _inFlightCount;

        public string             NodeId       { get; }
        public IProxyDataSource   Proxy        { get; }
        public int                Weight       { get; set; }
        public ProxyDataSourceRole NodeRole    { get; set; }
        public bool               IsAlive      { get; internal set; } = true;
        public DateTime           LastProbeUtc { get; internal set; } = DateTime.UtcNow;
        public int                InFlightCount => _inFlightCount;
        public DataSourceMetrics  Metrics      => Proxy.GetMetrics()?.Values
                                                      .FirstOrDefault() ?? new DataSourceMetrics();
        public DateTime           AddedAtUtc   { get; } = DateTime.UtcNow;

        // Internal: set by NodeProbing when node is draining
        internal bool IsDraining { get; set; }

        // Counters used by NodeProbing
        internal int ConsecutiveFailures  { get; set; }
        internal int ConsecutiveSuccesses { get; set; }

        public ProxyNode(string nodeId, IProxyDataSource proxy,
            int weight = 1,
            ProxyDataSourceRole role = ProxyDataSourceRole.Primary)
        {
            NodeId   = nodeId   ?? throw new ArgumentNullException(nameof(nodeId));
            Proxy    = proxy    ?? throw new ArgumentNullException(nameof(proxy));
            Weight   = weight;
            NodeRole = role;
        }

        internal void IncrementInFlight()  => Interlocked.Increment(ref _inFlightCount);
        internal void DecrementInFlight()  => Interlocked.Decrement(ref _inFlightCount);
    }
}
```

```csharp
            foreach (var n in _nodes.Values) n.Proxy?.Dispose();
        }
```

## 7. `INodeRouter` interface + all strategy classes

**File:** `ProxyCluster.NodeRouting.cs`

```csharp
internal interface INodeRouter
{
    /// <summary>Returns the best live node from <paramref name="candidates"/>.</summary>
    ProxyNode SelectBest(IReadOnlyList<ProxyNode> candidates, ProxyExecutionContext ctx = null);
}

// ── LeastConnections ──────────────────────────────────────────────────────
internal sealed class LeastConnectionsRouter : INodeRouter
{
    public ProxyNode SelectBest(IReadOnlyList<ProxyNode> candidates, ProxyExecutionContext ctx)
        => candidates.MinBy(n => n.InFlightCount);
}

// ── WeightedRoundRobin ────────────────────────────────────────────────────
internal sealed class WeightedRoundRobinRouter : INodeRouter
{
    private int _counter;
    public ProxyNode SelectBest(IReadOnlyList<ProxyNode> candidates, ProxyExecutionContext ctx)
    {
        // Build expanded list where each node appears Weight times
        var expanded = candidates.SelectMany(n => Enumerable.Repeat(n, Math.Max(1, n.Weight))).ToList();
        int idx = Interlocked.Increment(ref _counter) % expanded.Count;
        return expanded[idx];
    }
}

// ── LowestLatency ─────────────────────────────────────────────────────────
internal sealed class LowestLatencyRouter : INodeRouter
{
    public ProxyNode SelectBest(IReadOnlyList<ProxyNode> candidates, ProxyExecutionContext ctx)
        => candidates.MinBy(n => n.Metrics?.P50LatencyMs ?? double.MaxValue);
}

// ── PrimaryWithStandby ────────────────────────────────────────────────────
internal sealed class PrimaryWithStandbyRouter : INodeRouter
{
    public ProxyNode SelectBest(IReadOnlyList<ProxyNode> candidates, ProxyExecutionContext ctx)
    {
        var primary = candidates.FirstOrDefault(n => n.NodeRole == ProxyDataSourceRole.Primary);
        return primary ?? candidates.First();
    }
}

// ── ConsistentHash ────────────────────────────────────────────────────────
// Full design: 150 virtual slots per node (matches Redis Cluster convention).
// Hash function: MurmurHash3(correlationId) → uint → ring lookup.
// On topology change: only 1/N of keys remap.
internal sealed class ConsistentHashRouter : INodeRouter
{
    private SortedDictionary<uint, ProxyNode> _ring = new();
    private const int VirtualSlots = 150;

    internal void Rebuild(IEnumerable<ProxyNode> nodes)
    {
        _ring.Clear();
        foreach (var n in nodes)
            for (int v = 0; v < VirtualSlots; v++)
            {
                uint slot = MurmurHash3($"{n.NodeId}:{v}");
                _ring[slot] = n;
            }
    }

    public ProxyNode SelectBest(IReadOnlyList<ProxyNode> candidates, ProxyExecutionContext ctx)
    {
        if (_ring.Count == 0) return candidates.First();
        string key = ctx?.CorrelationId ?? Guid.NewGuid().ToString();
        uint hash = MurmurHash3(key);
        // Find the first ring slot >= hash; wrap around
        var entry = _ring.FirstOrDefault(kv => kv.Key >= hash);
        var node  = entry.Value ?? _ring.First().Value;
        // If the selected node is dead, walk clockwise
        if (!candidates.Contains(node))
            node = candidates.First();
        return node;
    }

    private static uint MurmurHash3(string s)
    {
        // ... standard MurmurHash3 32-bit implementation
    }
}
```

`ProxyCluster.SelectNode()` (in `ProxyCluster.cs`):

```csharp
private ProxyNode SelectNode(string operationName, string entityHint, ProxyExecutionContext ctx)
{
    var live = _nodes.Values
        .Where(n => n.IsAlive && !n.IsDraining)
        .ToList();

    if (live.Count == 0)
        throw new InvalidOperationException("ProxyCluster: no live nodes available.");

    // Slow-start: reduce effective weight for recently-added nodes
    if (_clusterPolicy.SlowStartDurationMs > 0)
        ApplySlowStartWeights(live);

    return _nodeRouter.SelectBest(live, ctx);
}
```

## 8. Slow-Start Weight Calculation

```csharp
private void ApplySlowStartWeights(List<ProxyNode> candidates)
{
    foreach (var n in candidates)
    {
        double elapsed   = (DateTime.UtcNow - n.AddedAtUtc).TotalMilliseconds;
        double rampFactor = Math.Min(1.0, elapsed / _clusterPolicy.SlowStartDurationMs);
        n.Weight = Math.Max(1, (int)(n.Weight * rampFactor));
    }
}
```
