Overview

The RegisterBeep class sets up the Beep Data Management Engine for direct data operations. Once configured, you work with DMEEditor to manage data sources, create connections, and perform data operations across multiple database types.

Purpose: RegisterBeep provides the foundation for data management - after setup, you use beepService.DMEEditor directly for all data operations.

Core Pattern

  1. Setup: Register Beep services and get IBeepService
  2. Connections: Create connection configurations for your databases
  3. Data Sources: Get data sources using DMEEditor.GetDataSource()
  4. Operations: Perform CRUD operations directly on data sources

Initial Setup

Basic Registration

// In your Program.cs or initialization code
var services = new ServiceCollection();
services.Register(AppContext.BaseDirectory, "MyApp", BeepConfigType.DataConnector);
var provider = services.BuildServiceProvider();

// Get the main service - this is your entry point
var beepService = provider.GetService<IBeepService>();

// DMEEditor is your main data management interface
var dmeEditor = beepService.DMEEditor;

Connection Management

Creating Database Connections

Based on the actual patterns from BeepSharedFunctions.cs:

SQLite Database Connection

public static bool CreateConnectionForSqlite()
{
    // Get the connection driver configuration
    ConnectionDriversConfig config = beepService.DMEEditor.ConfigEditor
        .DataDriversClasses.FirstOrDefault(p => p.DatasourceType == DataSourceType.SqlLite);
    
    // Create connection properties
    ConnectionProperties connectionProperties = new ConnectionProperties
    {
        ConnectionString = "Data Source=./Beep/dbfiles/northwind.db",
        ConnectionName = "northwind.db",
        DriverName = config.PackageName,
        DriverVersion = config.version,
        DatabaseType = DataSourceType.SqlLite,
        Category = DatasourceCategory.RDBMS
    };
    
    // Add to connection list
    beepService.Config_editor.AddDataConnection(connectionProperties);
    
    // Get and open the data source
    var sqliteDataSource = beepService.DMEEditor.GetDataSource("northwind.db");
    sqliteDataSource.Openconnection();
    
    return sqliteDataSource.ConnectionStatus == ConnectionState.Open;
}

Data Operations

Basic Data Access Pattern

// Get data source from DMEEditor
var dataSource = beepService.DMEEditor.GetDataSource("northwind.db");

// Ensure connection is open
if (dataSource != null)
{
    dataSource.Openconnection();
    
    if (dataSource.ConnectionStatus == ConnectionState.Open)
    {
        // Get list of entities (tables)
        var entities = dataSource.GetEntitesList();
        
        // Get data from specific entity
        var customerData = dataSource.GetEntity("Customers", null);
        
        // Get entity structure (schema information)
        var structure = dataSource.GetEntityStructure("Customers", true);
        
        // Use the data...
        ProcessData(customerData, structure);
    }
}

Data Movement Between Sources

From the real implementation in BeepSharedFunctions:

public static IErrorsInfo MoveEntity(string src, string dest, string entityname, 
    Progress<PassedArgs> progress)
{
    var sourceDataSource = beepService.DMEEditor.GetDataSource(src);
    var destDataSource = beepService.DMEEditor.GetDataSource(dest);
    
    if (sourceDataSource == null || destDataSource == null)
    {
        beepService.DMEEditor.AddLogMessage("Beep", "Source or Destination not found", 
            DateTime.Now, 0, "", Errors.Failed);
        return beepService.DMEEditor.ErrorObject;
    }
    
    // Open connections
    sourceDataSource.Openconnection();
    destDataSource.Openconnection();
    
    if (sourceDataSource.ConnectionStatus == ConnectionState.Open && 
        destDataSource.ConnectionStatus == ConnectionState.Open)
    {
        // Get entity structures
        EntityStructure srcEntity = sourceDataSource.GetEntityStructure(entityname, true);
        
        // Get data from source
        var data = sourceDataSource.GetEntity(entityname, null);
        
        // Create entity in destination if it doesn't exist
        if (!destDataSource.CheckEntityExist(entityname))
        {
            destDataSource.CreateEntityAs(srcEntity);
        }
        
        // Move the data
        destDataSource.UpdateEntities(entityname, data, progress);
    }
    
    return beepService.DMEEditor.ErrorObject;
}

Real Examples from Sample App

Form with Data Source Selection

public partial class UnitofWork : Form
{
    private IBeepService beepService;
    
    public UnitofWork(IBeepService bservice)
    {
        beepService = bservice;
        InitializeComponent();
        
        // Setup grid with Beep services
        beepGrid1.SetConfig(beepService.DMEEditor, beepService.lg, beepService.util, 
            new string[] { }, beepService.DMEEditor.Passedarguments, 
            beepService.DMEEditor.ErrorObject);
    }
    
    private void GetDataSourceFromCombobox()
    {
        if (dataSourcesComboBox.SelectedItem != null)
        {
            // Get data source directly from DMEEditor
            var dataSource = beepService.DMEEditor.GetDataSource(
                dataSourcesComboBox.SelectedItem.ToString());
            
            if (dataSource != null)
            {
                dataSource.Openconnection();
                
                if (dataSource.ConnectionStatus == ConnectionState.Open)
                {
                    LoadEntities(dataSource);
                }
                else
                {
                    beepService.DMEEditor.AddLogMessage("Beep", 
                        "Error Opening Connection", DateTime.Now, -1, null, Errors.Failed);
                }
            }
        }
    }
    
    private void LoadEntities(IDataSource dataSource)
    {
        entitiesComboBox.Items.Clear();
        var entities = dataSource.GetEntitesList();
        
        foreach (string entity in entities)
        {
            entitiesComboBox.Items.Add(entity);
        }
    }
}

Shared Functions for Connection Management

public static class BeepSharedFunctions
{
    public static IBeepService beepService;
    public static IDataSource SourceDataSource;
    public static IDataSource DestinationDataSource;
    
    public static ConnectionDriversConfig GetConnectionDriversConfig(DataSourceType type)
    {
        return beepService.DMEEditor.ConfigEditor.DataDriversClasses
            .FirstOrDefault(p => p.DatasourceType == type);
    }
    
    public static void CloseConnections()
    {
        SourceDataSource?.Closeconnection();
        DestinationDataSource?.Closeconnection();
    }
}

Best Practices

1. Always Check Connections

Always verify connection status before performing data operations. 
var dataSource = beepService.DMEEditor.GetDataSource("MyDatabase");
if (dataSource != null)
{
    dataSource.Openconnection();
    if (dataSource.ConnectionStatus == ConnectionState.Open)
    {
        // Safe to perform operations
    }
}

2. Use Proper Error Handling

try
{
    var data = dataSource.GetEntity("Customers", null);
    // Process data...
}
catch (Exception ex)
{
    beepService.DMEEditor.AddLogMessage("Beep", 
        $"Error loading data: {ex.Message}", 
        DateTime.Now, -1, null, Errors.Failed);
}

3. Clean Up Resources

Always close connections when done to free up resources. 
private void Form_Closing(object sender, FormClosingEventArgs e)
{
    // Close all data source connections
    BeepSharedFunctions.CloseConnections();
}

4. Use DMEEditor for All Data Operations

Access data sources through beepService.DMEEditor - this is the main interface for all data management operations.

5. Configuration First, Then Operations

Always create and add connection configurations before trying to get data sources. 
// 1. Create connection properties
var connectionProps = new ConnectionProperties { /* ... */ };

// 2. Add to configuration
beepService.Config_editor.AddDataConnection(connectionProps);

// 3. Get data source
var dataSource = beepService.DMEEditor.GetDataSource(connectionName);

// 4. Use data source
dataSource.Openconnection();
var data = dataSource.GetEntity("TableName", null);
