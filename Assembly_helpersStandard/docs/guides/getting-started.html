Getting Started

Quick start guide for implementing assembly loading and extension management

Prerequisites

Before getting started with the Assembly Helpers library, ensure you have:

.NET Framework

.NET 8 or .NET 9 target framework in your project

Development Environment

Visual Studio 2022 or VS Code with C# extensions

Permissions

File system access for assembly loading and scanning

Installation

NuGet Package

Install the Assembly Helpers package via NuGet Package Manager:

Package Manager 
Install-Package TheTechIdea.Beep.Assembly.Helpers

Or via .NET CLI:

.NET CLI 
dotnet add package TheTechIdea.Beep.Assembly.Helpers

Project References

Add the following using statements to your code files:

C# 
using TheTechIdea.Beep.Tools;
using TheTechIdea.Beep.Editor;
using TheTechIdea.Beep.Logger;
using TheTechIdea.Beep.Utilities;
using TheTechIdea.Beep.ConfigUtil;

Quick Start

Basic Setup

Create and initialize an AssemblyHandler instance with the required dependencies:

C# 
// Initialize dependencies
var configEditor = new ConfigEditor();
var errorObject = new ErrorsInfo();
var logger = new DMLogger();
var utilFunction = new Util();

// Create AssemblyHandler
var assemblyHandler = new AssemblyHandler(
    configEditor, 
    errorObject, 
    logger, 
    utilFunction
);

// Configure namespaces to ignore (optional)
assemblyHandler.NamespacestoIgnore.Add("System");
assemblyHandler.NamespacestoIgnore.Add("Microsoft");

Console.WriteLine("AssemblyHandler initialized successfully!");

Loading Assemblies

Load Single Directory

Load all assemblies from a specific directory:

C# 
// Load assemblies from a directory
string pluginPath = @"C:\MyApp\Plugins";
var result = assemblyHandler.LoadAssembly(pluginPath, FolderFileTypes.Addin);

if (assemblyHandler.ErrorObject.Flag == Errors.Ok)
{
    Console.WriteLine($"Successfully loaded assemblies from {pluginPath}");
    Console.WriteLine($"Total assemblies loaded: {assemblyHandler.Assemblies.Count}");
}
else
{
    Console.WriteLine($"Error loading assemblies: {assemblyHandler.ErrorObject.Message}");
}

Load All Configured Assemblies

Load assemblies from all configured folders with progress reporting:

C# 
// Setup progress reporting
var progress = new Progress<PassedArgs>(args => {
    Console.WriteLine($"[{DateTime.Now:HH:mm:ss}] {args.Messege}");
});

// Setup cancellation token
var cancellationTokenSource = new CancellationTokenSource();

// Load all assemblies
var result = assemblyHandler.LoadAllAssembly(progress, cancellationTokenSource.Token);

if (result.Flag == Errors.Ok)
{
    Console.WriteLine("All assemblies loaded successfully!");
    
    // Display summary
    Console.WriteLine($"Built-in assemblies: {assemblyHandler.Assemblies.Count(a => a.FileTypes == FolderFileTypes.Builtin)}");
    Console.WriteLine($"Plugin assemblies: {assemblyHandler.Assemblies.Count(a => a.FileTypes == FolderFileTypes.Addin)}");
    Console.WriteLine($"Data source assemblies: {assemblyHandler.Assemblies.Count(a => a.FileTypes == FolderFileTypes.DataSources)}");
}
else
{
    Console.WriteLine($"Error loading assemblies: {result.Message}");
}

Creating Instances

Create Instance by Type Name

Dynamically create instances of loaded types:

C# 
// Create instance without parameters
var instance = assemblyHandler.CreateInstanceFromString("MyNamespace.MyDataSource");

if (instance != null)
{
    Console.WriteLine($"Created instance of type: {instance.GetType().Name}");
    
    // Use the instance
    if (instance is IDataSource dataSource)
    {
        // Configure and use the data source
        Console.WriteLine($"Data source category: {dataSource.Category}");
    }
}

// Create instance with constructor parameters
var configuredInstance = assemblyHandler.CreateInstanceFromString(
    "MyNamespace.MyConfigurableService",
    "connectionString",
    true,
    1000
);

if (configuredInstance != null)
{
    Console.WriteLine("Successfully created configured instance");
}

Create Instance from Specific Assembly

Create instances from a specific DLL:

C# 
// Create instance from specific DLL
var instance = assemblyHandler.CreateInstanceFromString(
    "MyPlugin.dll",
    "MyNamespace.MyPluginClass",
    configParameter1,
    configParameter2
);

if (instance != null)
{
    Console.WriteLine($"Created instance from specific DLL: {instance.GetType().Name}");
}
else
{
    Console.WriteLine("Failed to create instance - check type name and assembly");
}

Working with Data Sources

Discover Data Sources

Find and work with discovered data source implementations:

C# 
// Get all discovered data sources
var dataSources = assemblyHandler.DataSourcesClasses;

Console.WriteLine($"Found {dataSources.Count} data source implementations:");

foreach (var ds in dataSources)
{
    Console.WriteLine($"- {ds.className} ({ds.componentType})");
    
    if (ds.classProperties != null)
    {
        Console.WriteLine($"  Category: {ds.classProperties.Category}");
        Console.WriteLine($"  Description: {ds.classProperties.Description}");
        
        if (!string.IsNullOrEmpty(ds.classProperties.FileType))
        {
            Console.WriteLine($"  Supported files: {ds.classProperties.FileType}");
        }
    }
}

// Filter by category
var fileSources = dataSources.Where(ds => 
    ds.classProperties?.Category == DatasourceCategory.FILE).ToList();

Console.WriteLine($"\nFile-based data sources: {fileSources.Count}");

// Create file extension list
var supportedExtensions = assemblyHandler.CreateFileExtensionString();
Console.WriteLine($"Supported file extensions: {string.Join(", ", supportedExtensions)}");

Work with Drivers

Discover and manage database drivers:

C# 
// Get available drivers from configuration
var drivers = assemblyHandler.ConfigEditor.DataDriversClasses;

Console.WriteLine($"Available drivers: {drivers.Count}");

foreach (var driver in drivers)
{
    Console.WriteLine($"- {driver.DriverClass} (v{driver.version})");
    Console.WriteLine($"  Package: {driver.PackageName}");
    Console.WriteLine($"  DLL: {driver.dllname}");
    
    if (driver.ADOType)
    {
        Console.WriteLine("  Type: ADO.NET Driver");
        if (!string.IsNullOrEmpty(driver.DbConnectionType))
            Console.WriteLine($"  Connection Type: {driver.DbConnectionType}");
    }
    else
    {
        Console.WriteLine("  Type: Custom Driver");
    }
    
    Console.WriteLine();
}

Error Handling

Check Operation Results

Properly handle errors and check operation results:

C# 
try
{
    // Attempt to load assemblies
    var result = assemblyHandler.LoadAssembly(@"C:\Plugins", FolderFileTypes.Addin);
    
    // Check the result
    if (assemblyHandler.ErrorObject.Flag == Errors.Ok)
    {
        Console.WriteLine("Assembly loading successful");
        
        // Log success details
        assemblyHandler.Logger.WriteLog("Assembly loading completed successfully");
    }
    else
    {
        Console.WriteLine($"Assembly loading failed: {assemblyHandler.ErrorObject.Message}");
        
        // Log error details
        assemblyHandler.Logger.WriteLog($"Assembly loading error: {assemblyHandler.ErrorObject.Message}");
        
        // Handle specific error cases
        switch (assemblyHandler.ErrorObject.Flag)
        {
            case Errors.Failed:
                Console.WriteLine("General failure occurred");
                break;
            case Errors.Warning:
                Console.WriteLine("Operation completed with warnings");
                break;
        }
    }
}
catch (DirectoryNotFoundException ex)
{
    Console.WriteLine($"Directory not found: {ex.Message}");
}
catch (UnauthorizedAccessException ex)
{
    Console.WriteLine($"Access denied: {ex.Message}");
}
catch (Exception ex)
{
    Console.WriteLine($"Unexpected error: {ex.Message}");
    assemblyHandler.Logger.WriteLog($"Unexpected error: {ex}");
}

Validation and Safety

Validate inputs and ensure safe operations:

C# 
// Validate directory before loading
string pluginDirectory = @"C:\MyApp\Plugins";

if (!Directory.Exists(pluginDirectory))
{
    Console.WriteLine($"Plugin directory does not exist: {pluginDirectory}");
    return;
}

// Check permissions
try
{
    var testFiles = Directory.GetFiles(pluginDirectory, "*.dll");
    Console.WriteLine($"Found {testFiles.Length} DLL files to process");
}
catch (UnauthorizedAccessException)
{
    Console.WriteLine("Insufficient permissions to access plugin directory");
    return;
}

// Safe type resolution
string typeName = "MyNamespace.MyClass";
var type = assemblyHandler.GetType(typeName);

if (type == null)
{
    Console.WriteLine($"Type '{typeName}' not found in loaded assemblies");
    
    // List available types for debugging
    var availableTypes = assemblyHandler.Assemblies
        .SelectMany(a => a.DllLib.GetTypes())
        .Where(t => t.Namespace?.Contains("MyNamespace") == true)
        .Select(t => t.FullName)
        .ToList();
        
    Console.WriteLine($"Available types in MyNamespace: {string.Join(", ", availableTypes)}");
}
else
{
    Console.WriteLine($"Type found: {type.FullName}");
}

Best Practices

Use Progress Reporting

Always implement progress reporting for long-running assembly loading operations to provide user feedback.

Configure Namespace Filtering

Set up NamespacestoIgnore to exclude system assemblies and improve scanning performance.

Handle Cancellation

Use CancellationToken for operations that might need to be cancelled by the user or system.

Validate Directory Paths

Always validate that directories exist and are accessible before attempting to load assemblies.

Check Error States

Always check ErrorObject.Flag after operations to handle failures gracefully.

Dispose Properly

AssemblyHandler implements IDisposable - ensure proper cleanup in using statements or try/finally blocks.

Complete Example

Here's a complete example that demonstrates the main features:

C# 
using System;
using System.IO;
using System.Linq;
using System.Threading;
using TheTechIdea.Beep.Tools;
using TheTechIdea.Beep.Editor;
using TheTechIdea.Beep.Logger;
using TheTechIdea.Beep.Utilities;

class Program
{
    static async Task Main(string[] args)
    {
        Console.WriteLine("Assembly Helpers Demo Application");
        Console.WriteLine("=================================");
        
        // Initialize dependencies
        using var assemblyHandler = CreateAssemblyHandler();
        
        // Setup progress reporting
        var progress = new Progress<PassedArgs>(args => {
            Console.WriteLine($"[Progress] {args.Messege}");
        });
        
        // Setup cancellation
        var cts = new CancellationTokenSource();
        Console.CancelKeyPress += (s, e) => {
            e.Cancel = true;
            cts.Cancel();
            Console.WriteLine("\nCancellation requested...");
        };
        
        try
        {
            // Load all assemblies
            Console.WriteLine("\nLoading assemblies...");
            var result = assemblyHandler.LoadAllAssembly(progress, cts.Token);
            
            if (result.Flag == Errors.Ok)
            {
                Console.WriteLine("\n? Assembly loading completed successfully!");
                
                // Display summary
                DisplaySummary(assemblyHandler);
                
                // Demonstrate instance creation
                DemonstrateInstanceCreation(assemblyHandler);
                
                // Show available data sources
                ShowDataSources(assemblyHandler);
            }
            else
            {
                Console.WriteLine($"\n? Assembly loading failed: {result.Message}");
            }
        }
        catch (OperationCanceledException)
        {
            Console.WriteLine("\nOperation was cancelled by user");
        }
        catch (Exception ex)
        {
            Console.WriteLine($"\nUnexpected error: {ex.Message}");
        }
        
        Console.WriteLine("\nPress any key to exit...");
        Console.ReadKey();
    }
    
    static AssemblyHandler CreateAssemblyHandler()
    {
        var configEditor = new ConfigEditor();
        var errorObject = new ErrorsInfo();
        var logger = new DMLogger();
        var utilFunction = new Util();
        
        var handler = new AssemblyHandler(configEditor, errorObject, logger, utilFunction);
        
        // Configure to ignore system namespaces
        handler.NamespacestoIgnore.AddRange(new[] {
            "System", "Microsoft", "Windows", "Mono"
        });
        
        return handler;
    }
    
    static void DisplaySummary(AssemblyHandler handler)
    {
        Console.WriteLine("\nAssembly Summary:");
        Console.WriteLine($"  Total assemblies: {handler.Assemblies.Count}");
        Console.WriteLine($"  Built-in: {handler.Assemblies.Count(a => a.FileTypes == FolderFileTypes.Builtin)}");
        Console.WriteLine($"  Data sources: {handler.Assemblies.Count(a => a.FileTypes == FolderFileTypes.DataSources)}");
        Console.WriteLine($"  Addins: {handler.Assemblies.Count(a => a.FileTypes == FolderFileTypes.Addin)}");
        Console.WriteLine($"  Extensions: {handler.LoaderExtensionClasses.Count}");
    }
    
    static void DemonstrateInstanceCreation(AssemblyHandler handler)
    {
        Console.WriteLine("\nDemonstrating instance creation...");
        
        // Try to create a common interface instance
        var dataSources = handler.DataSourcesClasses.Take(3);
        
        foreach (var ds in dataSources)
        {
            try
            {
                var instance = handler.CreateInstanceFromString(ds.PackageName);
                if (instance != null)
                {
                    Console.WriteLine($"? Created: {ds.className}");
                }
                else
                {
                    Console.WriteLine($"? Failed: {ds.className}");
                }
            }
            catch (Exception ex)
            {
                Console.WriteLine($"? Error creating {ds.className}: {ex.Message}");
            }
        }
    }
    
    static void ShowDataSources(AssemblyHandler handler)
    {
        Console.WriteLine("\nAvailable Data Sources:");
        
        var dataSources = handler.DataSourcesClasses
            .Where(ds => ds.classProperties != null)
            .GroupBy(ds => ds.classProperties.Category)
            .ToList();
            
        foreach (var group in dataSources)
        {
            Console.WriteLine($"\n  {group.Key}:");
            foreach (var ds in group.Take(5))
            {
                Console.WriteLine($"    - {ds.className}");
            }
        }
        
        // Show supported file extensions
        var extensions = handler.CreateFileExtensionString();
        if (extensions.Any())
        {
            Console.WriteLine($"\nSupported file extensions: {string.Join(", ", extensions.Take(10))}");
        }
    }
}
