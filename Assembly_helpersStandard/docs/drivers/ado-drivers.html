ADO.NET Drivers

Automatic discovery and configuration of ADO.NET database providers

Note

ADO.NET drivers provide connectivity to relational databases through the standard .NET data access framework. The AssemblyHandler automatically discovers and configures ADO.NET providers.

ADO.NET Driver Architecture

ADO.NET drivers in Beep follow the standard .NET data provider model with these core components:

C# 
// Core ADO.NET interfaces that drivers must implement
public interface IDbConnection : IDisposable
public interface IDbCommand : IDisposable  
public interface IDataReader : IDisposable, IDataRecord
public interface IDbDataAdapter
public interface IDbTransaction : IDisposable

// Base classes for providers
public abstract class DbConnection : IDbConnection
public abstract class DbCommand : IDbCommand
public abstract class DbDataReader : IDataReader
public abstract class DbDataAdapter : IDbDataAdapter
public abstract class DbTransaction : IDbTransaction

Automatic ADO.NET Discovery

Discovery Process

The GetADOTypeDrivers method automatically scans assemblies for ADO.NET implementations:

C# 
private bool GetADOTypeDrivers(Assembly asm)
{
    try
    {
        // Create thread-safe collection for driver configurations
        var driversConfigDict = new ConcurrentDictionary();

        // Extract assembly information
        string[] assemblyNameParts = asm.FullName.Split(',');
        string driverVersion = assemblyNameParts.Length > 1 
            ? assemblyNameParts[1].Substring(assemblyNameParts[1].IndexOf("=") + 1) 
            : "1.0";
        string packageName = assemblyNameParts[0];

        // Identify relevant ADO.NET types
        var relevantTypes = GetADONetTypes(asm);

        // Process types (parallel for large assemblies)
        if (relevantTypes.Count > 50)
        {
            Parallel.ForEach(relevantTypes, type =>
            {
                ProcessDriverType(type.GetTypeInfo(), packageName, driverVersion, 
                                type.Module.Name, driversConfigDict);
            });
        }
        else
        {
            foreach (var type in relevantTypes)
            {
                ProcessDriverType(type.GetTypeInfo(), packageName, driverVersion, 
                                type.Module.Name, driversConfigDict);
            }
        }

        // Add discovered drivers to main collection
        foreach (var driver in driversConfigDict.Values)
        {
            DataDriversConfig.Add(driver);
        }

        return true;
    }
    catch (Exception ex)
    {
        Logger?.WriteLog($"Failed to get ADO drivers from {asm.GetName()}", ex);
        return false;
    }
}

Type Processing

Each discovered ADO.NET type is processed to build the driver configuration:

private void ProcessDriverType(TypeInfo typeInfo, string packageName, string version, 
                                     string moduleName, ConcurrentDictionary driversConfigDict)
{
    // Get or create driver configuration
    var driverConfig = driversConfigDict.GetOrAdd(packageName, key => new ConnectionDriversConfig
    {
        PackageName = key,
        DriverClass = key,
        version = version,
        dllname = moduleName,
        ADOType = true
    });

    // Map specific ADO.NET types to configuration properties
    if (typeInfo.ImplementedInterfaces.Contains(typeof(IDbDataAdapter)))
    {
        driverConfig.AdapterType = typeInfo.FullName;
    }
    else if (typeInfo.BaseType?.ToString().Contains("DbCommandBuilder") == true)
    {
        driverConfig.CommandBuilderType = typeInfo.FullName;
    }
    else if (typeInfo.ImplementedInterfaces.Contains(typeof(IDbConnection)) || 
             typeof(DbConnection).IsAssignableFrom(typeInfo))
    {
        driverConfig.DbConnectionType = typeInfo.FullName;
    }
    else if (typeInfo.ImplementedInterfaces.Contains(typeof(IDbTransaction)) || 
             typeof(DbTransaction).IsAssignableFrom(typeInfo))
    {
        driverConfig.DbTransactionType = typeInfo.FullName;
    }
}

Driver Validation

ADO.NET drivers require specific validation to ensure proper implementation:

public bool ValidateADODriver(ConnectionDriversConfig driver)
{
    if (!driver.ADOType)
        return true; // Not an ADO driver
        
    try
    {
        var assembly = Assembly.LoadFrom(driver.dllname);
        
        // Validate connection type
        if (!string.IsNullOrEmpty(driver.DbConnectionType))
        {
            var connectionType = assembly.GetType(driver.DbConnectionType);
            if (connectionType == null)
            {
                Logger?.WriteLog($"Connection type not found: {driver.DbConnectionType}");
                return false;
            }
            
            if (!typeof(IDbConnection).IsAssignableFrom(connectionType))
            {
                Logger?.WriteLog($"Type does not implement IDbConnection: {driver.DbConnectionType}");
                return false;
            }
        }
        
        // Validate adapter type
        if (!string.IsNullOrEmpty(driver.AdapterType))
        {
            var adapterType = assembly.GetType(driver.AdapterType);
            if (adapterType == null)
            {
                Logger?.WriteLog($"Adapter type not found: {driver.AdapterType}");
                return false;
            }
            
            if (!typeof(IDbDataAdapter).IsAssignableFrom(adapterType))
            {
                Logger?.WriteLog($"Type does not implement IDbDataAdapter: {driver.AdapterType}");
                return false;
            }
        }
        
        // Validate command builder type
        if (!string.IsNullOrEmpty(driver.CommandBuilderType))
        {
            var builderType = assembly.GetType(driver.CommandBuilderType);
            if (builderType == null)
            {
                Logger?.WriteLog($"Command builder type not found: {driver.CommandBuilderType}");
                return false;
            }
        }
        
        // Validate transaction type
        if (!string.IsNullOrEmpty(driver.DbTransactionType))
        {
            var transactionType = assembly.GetType(driver.DbTransactionType);
            if (transactionType == null)
            {
                Logger?.WriteLog($"Transaction type not found: {driver.DbTransactionType}");
                return false;
            }
            
            if (!typeof(IDbTransaction).IsAssignableFrom(transactionType))
            {
                Logger?.WriteLog($"Type does not implement IDbTransaction: {driver.DbTransactionType}");
                return false;
            }
        }
        
        return true;
    }
    catch (Exception ex)
    {
        Logger?.WriteLog($"Error validating ADO driver {driver.DriverClass}: {ex.Message}");
        return false;
    }
}

Performance Optimizations

Parallel Processing

For assemblies with many types, parallel processing improves discovery performance:

// Use parallel processing for large assemblies
if (relevantTypes.Count > 50)
{
    // Process types in parallel
    Parallel.ForEach(relevantTypes, type =>
    {
        if (type.BaseType == null) return;
        
        TypeInfo typeInfo = type.GetTypeInfo();
        ProcessDriverType(typeInfo, packageName, driverVersion, 
                         type.Module.Name, driversConfigDict);
    });
}
else
{
    // Sequential processing for smaller assemblies
    foreach (var type in relevantTypes)
    {
        if (type.BaseType == null) continue;
        
        TypeInfo typeInfo = type.GetTypeInfo();
        ProcessDriverType(typeInfo, packageName, driverVersion, 
                         type.Module.Name, driversConfigDict);
    }
}

Troubleshooting ADO.NET Discovery

Common Issues

  • Missing Types: Assembly doesn't export required ADO.NET interfaces
  • Version Conflicts: Multiple versions of same provider loaded
  • Reflection Errors: Security restrictions preventing type access
  • Incomplete Drivers: Provider missing required components (adapter, command builder)

Diagnostic Methods

public void DiagnoseADODriverIssues(Assembly assembly)
{
    try
    {
        Logger?.WriteLog($"Diagnosing assembly: {assembly.FullName}");
        
        // Check if assembly exports types
        var exportedTypes = assembly.GetExportedTypes();
        Logger?.WriteLog($"Exported types count: {exportedTypes.Length}");
        
        // Check for ADO.NET interfaces
        var adoTypes = exportedTypes.Where(t => 
            typeof(IDbConnection).IsAssignableFrom(t) ||
            typeof(IDbDataAdapter).IsAssignableFrom(t) ||
            typeof(IDbCommand).IsAssignableFrom(t)).ToList();
            
        Logger?.WriteLog($"ADO.NET types found: {adoTypes.Count}");
        
        foreach (var type in adoTypes)
        {
            Logger?.WriteLog($"  - {type.FullName}");
            
            // Check constructors
            var constructors = type.GetConstructors();
            Logger?.WriteLog($"    Constructors: {constructors.Length}");
            
            // Check if instantiable
            try
            {
                var instance = Activator.CreateInstance(type);
                Logger?.WriteLog($"    Can instantiate: Yes");
            }
            catch (Exception ex)
            {
                Logger?.WriteLog($"    Can instantiate: No - {ex.Message}");
            }
        }
    }
    catch (Exception ex)
    {
        Logger?.WriteLog($"Diagnostic failed: {ex.Message}");
    }
}

Best Practices

ADO.NET Driver Best Practices

  • Provider Factory Pattern: Use DbProviderFactory when possible for better abstraction
  • Connection String Security: Never hardcode credentials in connection strings
  • Resource Management: Ensure proper disposal of connections, commands, and readers
  • Parameterized Queries: Always use parameterized queries to prevent SQL injection
  • Connection Pooling: Configure appropriate connection pooling settings
  • Error Handling: Implement database-specific error handling
  • Version Compatibility: Test with multiple versions of database providers
  • Performance Monitoring: Monitor connection and query performance
