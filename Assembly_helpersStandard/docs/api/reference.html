Common Usage Patterns

Basic Assembly Loading

// Initialize AssemblyHandler
var assemblyHandler = new AssemblyHandler(configEditor, errorObject, logger, util);

// Load all assemblies with progress reporting
var progress = new Progress<PassedArgs>(args => Console.WriteLine(args.Messege));
var result = assemblyHandler.LoadAllAssembly(progress, CancellationToken.None);

// Check result
if (result.Flag == Errors.Ok)
{
    Console.WriteLine("Assemblies loaded successfully");
    
    // Get discovered data sources
    var dataSources = assemblyHandler.DataSourcesClasses;
    foreach (var ds in dataSources)
    {
        Console.WriteLine($"Found: {ds.className}");
    }
}

Creating Type Instances

// Create instance from type name
var instance = assemblyHandler.CreateInstanceFromString(
    "MyNamespace.MyClass",
    constructorArg1,
    constructorArg2
);

// Get type information
var type = assemblyHandler.GetType("MyNamespace.MyClass");
if (type != null)
{
    var attributes = type.GetCustomAttributes();
    // Process attributes...
}

Driver Management

// Get available drivers
var drivers = configEditor.DataDriversClasses;

// Filter by category
var rdbmsDrivers = drivers
    .Where(d => d.DatasourceCategory == DatasourceCategory.RDBMS)
    .ToList();

// Create driver configuration
var newDriver = new ConnectionDriversConfig
{
    DriverClass = "MyDataSource",
    PackageName = "MyNamespace.MyDataSource",
    DatasourceCategory = DatasourceCategory.RDBMS,
    ADOType = false
};

configEditor.DataDriversClasses.Add(newDriver);
